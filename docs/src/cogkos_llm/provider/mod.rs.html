1//! LLM Provider Configuration Module
2//!
3//! 支持多类型 LLM 配置：text, embedding, image, audio, other
4//! 配置优先级：配置文件 > 环境变量 > 默认值
5
6pub mod config;
7
8use std::env;
9
10/// LLM 类型别名
11pub type LlmType = &'static str;
12
13/// LLM 配置类型
14pub const LLM_TYPES: &[&str] = &["text", "embedding", "image", "audio", "other"];
15
16/// 环境变量映射到 LLM 类型
17pub fn env_key_for_llm_type(llm_type: &str) -> Option<&'static str> {
18    match llm_type {
19        "text" => Some("KIMI_API_KEY"),
20        "embedding" => Some("AI302_API_KEY"),
21        "image" => Some("DOUBAO_API_KEY"),
22        "audio" => Some("OPENAI_API_KEY"),
23        "other" => Some("OPENROUTER_API_KEY"),
24        _ => None,
25    }
26}
27
28/// 从环境变量获取 API Key
29pub fn get_api_key_from_env(llm_type: &str) -> Option<String> {
30    env_key_for_llm_type(llm_type)
31        .and_then(|key| env::var(key).ok())
32        .filter(|v| !v.is_empty())
33}
