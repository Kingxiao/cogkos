51    /// Train a new model
52    pub fn train(
53        &self,
54        domain: &str,
55        samples: &[TrainingSample],
56    ) -> Result<DedicatedPredictionModel> {
57        // Check data sufficiency
58        let analyzer = DataSufficiencyAnalyzer::new();
59        let sufficiency = analyzer.analyze(samples);
60
61        if !sufficiency.is_sufficient {
62            return Err(DedicatedModelError::InsufficientData(format!(
63                "Data insufficient: {:?}",
64                sufficiency.gaps
65            )));
66        }
67
68        let model_id = format!("{}-{}", domain, Uuid::new_v4());
69        let now = Utc::now();
70
71        // Simulate training
72        let training_info = TrainingInfo {
73            started_at: now,
74            completed_at: Some(now + Duration::minutes(30)),
75            sample_count: samples.len(),
76            training_duration_seconds: 1800,
77            data_hash: format!("hash_{}", samples.len()),
78            validation_split: self.config.validation_split,
79        };
80
81        // Simulate performance based on data quality
82        let base_performance = sufficiency.estimated_performance;
83        let performance = ModelPerformance {
84            accuracy: base_performance * 0.9 + 0.05,
85            precision: base_performance * 0.88 + 0.06,
86            recall: base_performance * 0.87 + 0.07,
87            f1_score: base_performance * 0.875 + 0.065,
88            confidence_calibration: 0.8,
89            latency_ms: match self.default_architecture {
90                ModelArchitecture::FineTunedSmall => 50,
91                ModelArchitecture::FineTunedMedium => 100,
92                ModelArchitecture::FineTunedLarge => 200,
93                ModelArchitecture::LoraAdapter => 75,
94                ModelArchitecture::PromptTuned => 60,
95                ModelArchitecture::RagEnhanced => 150,
96            },
97            evaluated_at: Utc::now(),
98        };
99
100        Ok(DedicatedPredictionModel {
101            model_id,
102            domain: domain.to_string(),
103            architecture: self.default_architecture,
104            status: ModelStatus::Ready,
105            version: 1,
106            training_info,
107            performance,
108            created_at: now,
109            updated_at: now,
110            hyperparameters: {
111                let mut params = HashMap::new();
112                params.insert("learning_rate".to_string(), self.config.learning_rate);
113                params.insert("epochs".to_string(), self.config.epochs as f64);
114                params
115            },
116        })
117    }

119    /// Retrain existing model with new data
120    pub fn retrain(
121        &self,
122        existing_model: &DedicatedPredictionModel,
123        new_samples: &[TrainingSample],
124    ) -> Result<DedicatedPredictionModel> {
125        let mut model = existing_model.clone();
126        model.version += 1;
127        model.status = ModelStatus::Training;
128
129        // Simulate retraining
130        model.training_info.started_at = Utc::now();
131        model.training_info.completed_at = Some(Utc::now() + Duration::minutes(20));
132        model.training_info.sample_count = new_samples.len();
133        model.updated_at = Utc::now();
134        model.status = ModelStatus::Ready;
135
136        // Slightly improve performance
137        model.performance.accuracy = (model.performance.accuracy * 1.02).min(0.95);
138        model.performance.evaluated_at = Utc::now();
139
140        Ok(model)
141    }
142}

177/// Model switch decision
178#[derive(Debug, Clone)]
179pub struct ModelSwitchDecision {
180    pub should_switch: bool,
181    pub from_model: String,
182    pub to_model: String,
183    pub improvement: f64,
184    pub confidence: f64,
185    pub risk_level: SwitchRiskLevel,
186    pub recommendation: SwitchRecommendation,
187}

189#[derive(Debug, Clone, Copy, PartialEq, Eq)]
190pub enum SwitchRiskLevel {
191    Low,
192    Medium,
193    High,
194}

196#[derive(Debug, Clone, Copy, PartialEq, Eq)]
197pub enum SwitchRecommendation {
198    SwitchImmediately,
199    GradualTransition,
200    AwaitMoreData,
201    DoNotSwitch,
202    Rollback,
203}

255    /// Evaluate whether to switch models
256    pub fn evaluate_switch(
257        &self,
258        current_model: &DedicatedPredictionModel,
259        candidate_model: &DedicatedPredictionModel,
260    ) -> ModelSwitchDecision {
261        let improvement = candidate_model.performance.accuracy - current_model.performance.accuracy;
262        let relative_improvement = if current_model.performance.accuracy > 0.0 {
263            improvement / current_model.performance.accuracy
264        } else {
265            0.0
266        };
267
268        // Determine if improvement is significant
269        let is_significant = relative_improvement >= self.config.min_improvement_threshold;
270
271        // Calculate confidence based on sample sizes and consistency
272        let confidence = (candidate_model.training_info.sample_count as f64 / 1000.0).min(1.0)
273            * 0.7
274            + candidate_model.performance.confidence_calibration * 0.3;
275
276        // Determine risk level
277        let risk_level = if relative_improvement > 0.2 {
278            SwitchRiskLevel::High // Large jump is risky
279        } else if relative_improvement > 0.1 {
280            SwitchRiskLevel::Medium
281        } else {
282            SwitchRiskLevel::Low
283        };
284
285        // Make recommendation
286        let (should_switch, recommendation) = if !is_significant {
287            (false, SwitchRecommendation::DoNotSwitch)
288        } else if self.config.require_significance && confidence < 0.7 {
289            (false, SwitchRecommendation::AwaitMoreData)
290        } else if risk_level == SwitchRiskLevel::High {
291            (true, SwitchRecommendation::GradualTransition)
292        } else {
293            (true, SwitchRecommendation::SwitchImmediately)
294        };
295
296        ModelSwitchDecision {
297            should_switch,
298            from_model: current_model.model_id.clone(),
299            to_model: candidate_model.model_id.clone(),
300            improvement: relative_improvement,
301            confidence,
302            risk_level,
303            recommendation,
304        }
305    }

307    /// Execute a model switch
308    pub fn execute_switch(
309        &mut self,
310        domain: &str,
311        from_model: &str,
312        to_model: &str,
313    ) -> Result<PendingSwitch> {
314        let now = Utc::now();
315        let completion = now + Duration::minutes(self.config.transition_period_minutes);
316
317        let pending = PendingSwitch {
318            domain: domain.to_string(),
319            from_model: from_model.to_string(),
320            to_model: to_model.to_string(),
321            initiated_at: now,
322            completion_at: completion,
323            status: SwitchStatus::Pending,
324        };
325
326        self.pending_switches.push(pending.clone());
327
328        // Update active model
329        self.active_models
330            .insert(domain.to_string(), to_model.to_string());
331
332        Ok(pending)
333    }

335    /// Rollback to previous model
336    pub fn rollback(&mut self, domain: &str) -> Result<()> {
337        let current = self
338            .active_models
339            .get(domain)
340            .ok_or_else(|| DedicatedModelError::SwitchFailed("No active model".to_string()))?;
341
342        // Find the pending switch for this domain
343        if let Some(pending) = self
344            .pending_switches
345            .iter_mut()
346            .find(|s| s.domain == domain && s.to_model == *current)
347        {
348            pending.status = SwitchStatus::RolledBack;
349            self.active_models
350                .insert(domain.to_string(), pending.from_model.clone());
351            Ok(())
352        } else {
353            Err(DedicatedModelError::SwitchFailed(
354                "No switch to rollback".to_string(),
355            ))
356        }
357    }

362    }

364    /// Check if should auto-rollback based on performance
365    pub fn check_auto_rollback(
366        &self,
367        current_performance: &ModelPerformance,
368        baseline_performance: &ModelPerformance,
369    ) -> bool {
370        if !self.config.enable_auto_rollback {
371            return false;
372        }
373
374        let degradation = baseline_performance.accuracy - current_performance.accuracy;
375        let relative_degradation = if baseline_performance.accuracy > 0.0 {
376            degradation / baseline_performance.accuracy
377        } else {
378            0.0
379        };
380
381        relative_degradation > self.config.rollback_threshold
382    }

391/// Complete dedicated model system
392pub struct DedicatedModelSystem {
393    pub analyzer: DataSufficiencyAnalyzer,
394    pub trainer: ModelTrainer,
395    pub switch_manager: ModelSwitchManager,
396    pub models: HashMap<String, DedicatedPredictionModel>,
397}
398
399impl DedicatedModelSystem {
400    pub fn new() -> Self {
401        Self {
402            analyzer: DataSufficiencyAnalyzer::new(),
403            trainer: ModelTrainer::new(),
404            switch_manager: ModelSwitchManager::new(),
405            models: HashMap::new(),
406        }
407    }
408
409    /// Analyze data sufficiency for a domain
410    pub fn analyze_data(&self, samples: &[TrainingSample]) -> DataSufficiencyResult {
411        self.analyzer.analyze(samples)
412    }
413
414    /// Train a new model for a domain
415    pub fn train_model(
416        &mut self,
417        domain: &str,
418        samples: &[TrainingSample],
419    ) -> Result<DedicatedPredictionModel> {
420        let model = self.trainer.train(domain, samples)?;
421        self.models.insert(model.model_id.clone(), model.clone());
422        Ok(model)
423    }
424
425    /// Evaluate and potentially switch models
426    pub fn evaluate_and_switch(
427        &mut self,
428        domain: &str,
429        current_model_id: &str,
430        candidate_model: &DedicatedPredictionModel,
431    ) -> Option<ModelSwitchDecision> {
432        let current = self.models.get(current_model_id)?;
433        let decision = self
434            .switch_manager
435            .evaluate_switch(current, candidate_model);
436
437        if decision.should_switch {
438            self.switch_manager
439                .execute_switch(domain, current_model_id, &candidate_model.model_id)
440                .ok()?;
441        }
442
443        Some(decision)
444    }
445
446    /// Get active model for domain
447    pub fn get_active_model(&self, domain: &str) -> Option<&DedicatedPredictionModel> {
448        let model_id = self.switch_manager.get_active_model(domain)?;
449        self.models.get(model_id)
450    }
451}

453impl Default for DedicatedModelSystem {
454    fn default() -> Self {
455        Self::new()
