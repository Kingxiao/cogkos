143        workflow_id: String,
144        execution_id: String,
145        initial_vars: HashMap<String, Value>,
146    },
147    CancelExecution {

158impl WorkflowEngine {
159    pub fn new() -> Self {
160        let (command_tx, mut command_rx) = mpsc::unbounded_channel();
161        let engine = Self {
162            workflows: Arc::new(DashMap::new()),
163            executions: Arc::new(DashMap::new()),
164            handlers: Arc::new(DashMap::new()),
165            command_tx,
166        };
167
168        // Spawn command processing loop
169        let workflows = engine.workflows.clone();
170        let executions = engine.executions.clone();
171        tokio::spawn(async move {
172            while let Some(cmd) = command_rx.recv().await {
173                match cmd {
174                    EngineCommand::StartExecution {
175                        workflow_id,
176                        execution_id,
177                        initial_vars,
178                    } => {
179                        if let Some(_workflow) = workflows.get(&workflow_id) {
180                            let mut state = WorkflowState::new(execution_id.clone());
181                            state.variables = initial_vars;
182                            state.status = ExecutionStatus::Running;
183                            executions.insert(execution_id.clone(), Arc::new(RwLock::new(state)));
184                            info!("Started workflow execution: {}", execution_id);
185                        } else {
186                            error!("Workflow not found: {}", workflow_id);
187                        }
188                    }
189                    EngineCommand::CancelExecution { execution_id } => {
190                        if let Some(exec) = executions.get(&execution_id) {
191                            let mut state = exec.write().await;
192                            state.status = ExecutionStatus::Cancelled;
193                            info!("Cancelled workflow execution: {}", execution_id);
194                        }
195                    }
196                    EngineCommand::PauseExecution { execution_id } => {
197                        if let Some(exec) = executions.get(&execution_id) {
198                            let mut state = exec.write().await;
199                            if state.status == ExecutionStatus::Running {
200                                state.status = ExecutionStatus::Paused;
201                                info!("Paused workflow execution: {}", execution_id);
202                            }
203                        }
204                    }
205                    EngineCommand::ResumeExecution { execution_id } => {
206                        if let Some(exec) = executions.get(&execution_id) {
207                            let mut state = exec.write().await;
208                            if state.status == ExecutionStatus::Paused {
209                                state.status = ExecutionStatus::Running;
210                                info!("Resumed workflow execution: {}", execution_id);
211                            }
212                        }
213                    }
214                }
215            }
216        });
217
218        engine
219    }

307    pub async fn execute_workflow(&self, execution_id: &str) -> Result<WorkflowState> {
308        let execution = self
309            .engine
310            .executions
311            .get(execution_id)
312            .ok_or_else(|| WorkflowError::ExecutionError("Execution not found".to_string()))?;
313
314        // Get workflow ID from state
315        let workflow_id = {
316            let state = execution.read().await;
317            state.workflow_id.clone()
318        };
319
320        let workflow = self
321            .engine
322            .workflows
323            .get(&workflow_id)
324            .ok_or_else(|| WorkflowError::WorkflowNotFound(workflow_id.clone()))?;
325
326        // BFS execution
327        let mut queue = VecDeque::new();
328        queue.push_back(workflow.start_node);
329
330        while let Some(node_idx) = queue.pop_front() {
331            // Check if execution is still running
332            {
333                let state = execution.read().await;
334                if state.status != ExecutionStatus::Running {
335                    break;
336                }
337            }
338
339            let node = &workflow.graph[node_idx];
340
341            // Execute node
342            let result = self.execute_node(&workflow_id, execution_id, node).await;
343
344            match result {
345                Ok(node_result) => {
346                    let mut state = execution.write().await;
347                    state.completed_nodes.insert(node.id.clone());
348
349                    // Store output in variables
350                    if !node_result.output.is_null() {
351                        state.set_variable(format!("output.{}", node.id), node_result.output);
352                    }
353
354                    // Add next nodes to queue
355                    for edge in workflow.graph.edges(node_idx) {
356                        let target = edge.target();
357                        let edge_type = edge.weight();
358
359                        match edge_type {
360                            EdgeType::Sequential => {
361                                if !state.completed_nodes.contains(&workflow.graph[target].id) {
362                                    queue.push_back(target);
363                                }
364                            }
365                            EdgeType::Conditional { condition } => {
366                                if self.evaluate_condition(condition, &state) {
367                                    queue.push_back(target);
368                                }
369                            }
370                            EdgeType::Error => {}
371                        }
372                    }
373                }
374                Err(e) => {
375                    let mut state = execution.write().await;
376                    state.failed_nodes.insert(node.id.clone());
377                    state.status = ExecutionStatus::Failed;
378                    error!("Node execution failed: {:?}", e);
379                    break;
380                }
381            }
382        }
383
384        // Mark as completed if all nodes processed
385        let mut state = execution.write().await;
386        if state.status == ExecutionStatus::Running {
387            state.status = ExecutionStatus::Completed;
388            state.completed_at = Some(chrono::Utc::now());
389        }
390
391        Ok(state.clone())
392    }

394    async fn execute_node(
395        &self,
396        _workflow_id: &str,
397        execution_id: &str,
398        node: &WorkflowNode,
399    ) -> Result<NodeResult> {
400        let start = std::time::Instant::now();
401
402        // Get execution context
403        let ctx = self.build_context(execution_id, node).await?;
404
405        // Execute based on node type
406        let result = match &node.node_type {
407            NodeType::Task => {
408                if let Some(handler) = node.handler.as_ref() {
409                    handler(ctx).await
410                } else if let Some(handler) = self.engine.handlers.get("task") {
411                    handler(ctx).await
412                } else {
413                    Err(WorkflowError::ExecutionError(
414                        "No handler for task node".to_string(),
415                    ))
416                }
417            }
418            NodeType::Condition => self.execute_condition_node(ctx, node).await,
419            NodeType::Parallel => self.execute_parallel_node(ctx, node).await,
420            NodeType::Wait => self.execute_wait_node(ctx, node).await,
421            NodeType::Map => self.execute_map_node(ctx, node).await,
422            NodeType::Reduce => self.execute_reduce_node(ctx, node).await,
423            NodeType::SubWorkflow => self.execute_subworkflow_node(ctx, node).await,
424            // Evolution engine nodes
425            NodeType::BayesianAggregate => self.execute_bayesian_aggregate_node(ctx, node).await,
426            NodeType::KnowledgeDecay => self.execute_knowledge_decay_node(ctx, node).await,
427            NodeType::ConflictDetection => self.execute_conflict_detection_node(ctx, node).await,
428            NodeType::Revalidation => self.execute_revalidation_node(ctx, node).await,
429        };
430
431        let duration_ms = start.elapsed().as_millis() as u64;
432
433        match result {
434            Ok(mut r) => {
435                r.duration_ms = duration_ms;
436                Ok(r)
437            }
438            Err(e) => Ok(NodeResult {
439                node_id: node.id.clone(),
440                success: false,
441                output: Value::Null,
442                duration_ms,
443                error: Some(e.to_string()),
444            }),
445        }
446    }

496    async fn execute_parallel_node(
497        &self,
498        ctx: ExecutionContext,
499        node: &WorkflowNode,
500    ) -> Result<NodeResult> {
501        let branches = node
502            .config
503            .get("branches")
504            .and_then(|b| b.as_array())
505            .ok_or_else(|| WorkflowError::ExecutionError("Missing branches".to_string()))?;
506
507        let mut handles = vec![];
508
509        for (i, _branch) in branches.iter().enumerate() {
510            let _ctx_clone = ctx.clone();
511            let branch_id = format!("{}_branch_{}", node.id, i);
512
513            let handle = tokio::spawn(async move {
514                // Simulate branch execution
515                tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
516                serde_json::json!({ "branch_id": branch_id, "result": "completed" })
517            });
518
519            handles.push(handle);
520        }
521
522        let results: Vec<_> = futures::future::join_all(handles)
523            .await
524            .into_iter()
525            .filter_map(|r| r.ok())
526            .collect();
527
528        Ok(NodeResult {
529            node_id: node.id.clone(),
530            success: true,
531            output: serde_json::json!({ "branches": results }),
532            duration_ms: 0,
533            error: None,
534        })
535    }

559    /// Execute a Map node - applies operation to each element in a collection
560    async fn execute_map_node(
561        &self,
562        ctx: ExecutionContext,
563        node: &WorkflowNode,
564    ) -> Result<NodeResult> {
565        // Get the collection to iterate over
566        let collection_key = node
567            .config
568            .get("collection")
569            .and_then(|c| c.as_str())
570            .ok_or_else(|| WorkflowError::ExecutionError("Missing collection key".to_string()))?;
571
572        let collection = ctx
573            .variables
574            .get(collection_key)
575            .or_else(|| ctx.inputs.get(collection_key))
576            .and_then(|v| v.as_array())
577            .cloned()
578            .ok_or_else(|| {
579                WorkflowError::ExecutionError(format!("Collection not found: {}", collection_key))
580            })?;
581
582        // Get the operation to apply (handler name or inline expression)
583        let operation = node
584            .config
585            .get("operation")
586            .and_then(|o| o.as_str())
587            .unwrap_or("map");
588
589        // Get variable name to store results
590        let output_key = node
591            .config
592            .get("output")
593            .and_then(|o| o.as_str())
594            .unwrap_or("map_result");
595
596        let mut results = Vec::new();
597
598        // Execute operation for each element
599        for (i, item) in collection.iter().enumerate() {
600            let item_ctx = ExecutionContext {
601                workflow_id: ctx.workflow_id.clone(),
602                node_id: format!("{}_item_{}", node.id, i),
603                variables: ctx.variables.clone(),
604                inputs: [
605                    ("item".to_string(), item.clone()),
606                    ("index".to_string(), serde_json::json!(i)),
607                    ("operation".to_string(), serde_json::json!(operation)),
608                ]
609                .into_iter()
610                .collect(),
611            };
612
613            // Try custom handler first, then built-in operations
614            let result = if let Some(handler) = node.handler.as_ref() {
615                handler(item_ctx).await
616            } else if let Some(handler) = self.engine.handlers.get(operation) {
617                handler(item_ctx).await
618            } else {
619                // Built-in map operations
620                let item_val = item_ctx.get_input("item").cloned().unwrap_or(Value::Null);
621                Ok(NodeResult {
622                    node_id: item_ctx.node_id.clone(),
623                    success: true,
624                    output: item_val,
625                    duration_ms: 0,
626                    error: None,
627                })
628            };
629
630            match result {
631                Ok(r) => results.push(r.output),
632                Err(e) => {
633                    return Err(WorkflowError::ExecutionError(format!(
634                        "Map operation failed at index {}: {}",
635                        i, e
636                    )));
637                }
638            }
639        }
640
641        let output = serde_json::json!({
642            output_key: results,
643            "count": results.len()
644        });
645
646        Ok(NodeResult {
647            node_id: node.id.clone(),
648            success: true,
649            output,
650            duration_ms: 0,
651            error: None,
652        })
653    }

655    /// Execute a Reduce node - aggregates a collection into a single value
656    async fn execute_reduce_node(
657        &self,
658        ctx: ExecutionContext,
659        node: &WorkflowNode,
660    ) -> Result<NodeResult> {
661        // Get the collection to reduce
662        let collection_key = node
663            .config
664            .get("collection")
665            .and_then(|c| c.as_str())
666            .ok_or_else(|| WorkflowError::ExecutionError("Missing collection key".to_string()))?;
667
668        let collection = ctx
669            .variables
670            .get(collection_key)
671            .or_else(|| ctx.inputs.get(collection_key))
672            .and_then(|v| v.as_array())
673            .cloned()
674            .ok_or_else(|| {
675                WorkflowError::ExecutionError(format!("Collection not found: {}", collection_key))
676            })?;
677
678        // Get the reduction operation
679        let operation = node
680            .config
681            .get("operation")
682            .and_then(|o| o.as_str())
683            .unwrap_or("sum");
684
685        // Get initial value if specified
686        let initial_value = node.config.get("initial").cloned();
687
688        // Get variable name to store result
689        let output_key = node
690            .config
691            .get("output")
692            .and_then(|o| o.as_str())
693            .unwrap_or("reduce_result");
694
695        // Perform reduction
696        let result = match operation {
697            "sum" => {
698                let sum: f64 = collection.iter().filter_map(|v| v.as_f64()).sum();
699                serde_json::json!(sum)
700            }
701            "avg" | "average" => {
702                let sum: f64 = collection.iter().filter_map(|v| v.as_f64()).sum();
703                let count = collection.iter().filter_map(|v| v.as_f64()).count();
704                if count > 0 {
705                    serde_json::json!(sum / count as f64)
706                } else {
707                    serde_json::json!(0)
708                }
709            }
710            "min" => collection
711                .iter()
712                .filter_map(|v| v.as_f64())
713                .reduce(f64::min)
714                .map(|v| serde_json::json!(v))
715                .unwrap_or(serde_json::json!(null)),
716            "max" => collection
717                .iter()
718                .filter_map(|v| v.as_f64())
719                .reduce(f64::max)
720                .map(|v| serde_json::json!(v))
721                .unwrap_or(serde_json::json!(null)),
722            "count" => serde_json::json!(collection.len()),
723            "collect" => serde_json::json!(collection),
724            "merge" | "concat" => {
725                let mut strings = Vec::new();
726                for item in &collection {
727                    if let Some(s) = item.as_str() {
728                        strings.push(s.to_string());
729                    } else if let Some(arr) = item.as_array() {
730                        for v in arr {
731                            if let Some(s) = v.as_str() {
732                                strings.push(s.to_string());
733                            }
734                        }
735                    }
736                }
737                serde_json::json!(strings.join(""))
738            }
739            "custom" => {
740                // Custom reduction via handler
741                let mut accumulator = initial_value.unwrap_or(Value::Null);
742
743                for (i, item) in collection.iter().enumerate() {
744                    let reduce_ctx = ExecutionContext {
745                        workflow_id: ctx.workflow_id.clone(),
746                        node_id: format!("{}_reduce_{}", node.id, i),
747                        variables: ctx.variables.clone(),
748                        inputs: [
749                            ("accumulator".to_string(), accumulator),
750                            ("item".to_string(), item.clone()),
751                            ("index".to_string(), serde_json::json!(i)),
752                        ]
753                        .into_iter()
754                        .collect(),
755                    };
756
757                    if let Some(handler) = node.handler.as_ref() {
758                        accumulator = handler(reduce_ctx).await?.output;
759                    } else {
760                        return Err(WorkflowError::ExecutionError(
761                            "Custom reduce requires a handler".to_string(),
762                        ));
763                    }
764                }
765                accumulator
766            }
767            _ => {
768                return Err(WorkflowError::ExecutionError(format!(
769                    "Unknown reduce operation: {}",
770                    operation
771                )));
772            }
773        };
774
775        let output = serde_json::json!({
776            output_key: result,
777            "operation": operation,
778            "count": collection.len()
779        });
780
781        Ok(NodeResult {
782            node_id: node.id.clone(),
783            success: true,
784            output,
785            duration_ms: 0,
786            error: None,
787        })
788    }

790    /// Execute a SubWorkflow node - invokes a child workflow
791    async fn execute_subworkflow_node(
792        &self,
793        ctx: ExecutionContext,
794        node: &WorkflowNode,
795    ) -> Result<NodeResult> {
796        // Get the sub-workflow ID to execute
797        let workflow_id = node
798            .config
799            .get("workflow_id")
800            .and_then(|w| w.as_str())
801            .ok_or_else(|| WorkflowError::ExecutionError("Missing workflow_id".to_string()))?;
802
803        // Get input mappings (how to pass data to sub-workflow)
804        let input_mapping = node
805            .config
806            .get("inputs")
807            .and_then(|i| i.as_object())
808            .map(|o| {
809                o.iter()
810                    .map(|(k, v)| (k.clone(), v.clone()))
811                    .collect::<HashMap<String, Value>>()
812            })
813            .unwrap_or_default();
814
815        // Build input variables for sub-workflow
816        let mut sub_vars = HashMap::new();
817        for (target_key, source_expr) in input_mapping {
818            // Simple variable reference resolution
819            let source_key = source_expr
820                .as_str()
821                .map(|s| s.to_string())
822                .unwrap_or_else(|| source_expr.to_string());
823            if let Some(source_val) = ctx.variables.get(&source_key) {
824                sub_vars.insert(target_key.clone(), source_val.clone());
825            } else if let Some(source_val) = ctx.inputs.get(&source_key) {
826                sub_vars.insert(target_key.clone(), source_val.clone());
827            } else {
828                // Use literal value
829                sub_vars.insert(target_key.clone(), source_expr.clone());
830            }
831        }
832
833        // Start sub-workflow execution
834        let execution_id = self.engine.start_execution(workflow_id, sub_vars)?;
835
836        // Wait for sub-workflow to complete (with timeout)
837        let timeout_ms = node
838            .config
839            .get("timeout_ms")
840            .and_then(|t| t.as_u64())
841            .unwrap_or(30000); // Default 30s timeout
842
843        let result = tokio::time::timeout(tokio::time::Duration::from_millis(timeout_ms), async {
844            loop {
845                if let Some(state) = self.engine.get_execution_state(&execution_id).await {
846                    match state.status {
847                        ExecutionStatus::Completed => {
848                            break Ok(state);
849                        }
850                        ExecutionStatus::Failed | ExecutionStatus::Cancelled => {
851                            let status_str = match state.status {
852                                ExecutionStatus::Pending => "pending",
853                                ExecutionStatus::Running => "running",
854                                ExecutionStatus::Completed => "completed",
855                                ExecutionStatus::Failed => "failed",
856                                ExecutionStatus::Cancelled => "cancelled",
857                                ExecutionStatus::Paused => "paused",
858                            };
859                            break Err(WorkflowError::ExecutionError(format!(
860                                "Sub-workflow {} {}",
861                                status_str, execution_id
862                            )));
863                        }
864                        _ => {
865                            tokio::time::sleep(tokio::time::Duration::from_millis(50)).await;
866                        }
867                    }
868                } else {
869                    break Err(WorkflowError::ExecutionError(format!(
870                        "Sub-workflow execution not found: {}",
871                        execution_id
872                    )));
873                }
874            }
875        })
876        .await;
877
878        match result {
879            Ok(Ok(state)) => {
880                let output_key = node
881                    .config
882                    .get("output")
883                    .and_then(|o| o.as_str())
884                    .unwrap_or("subworkflow_result");
885
886                let output = serde_json::json!({
887                    output_key: {
888                        "execution_id": execution_id,
889                        "status": match state.status {
890                            ExecutionStatus::Pending => "pending",
891                            ExecutionStatus::Running => "running",
892                            ExecutionStatus::Completed => "completed",
893                            ExecutionStatus::Failed => "failed",
894                            ExecutionStatus::Cancelled => "cancelled",
895                            ExecutionStatus::Paused => "paused",
896                        },
897                        "variables": state.variables
898                    }
899                });
900
901                Ok(NodeResult {
902                    node_id: node.id.clone(),
903                    success: state.status == ExecutionStatus::Completed,
904                    output,
905                    duration_ms: 0,
906                    error: None,
907                })
908            }
909            Ok(Err(e)) => Err(e),
910            Err(_) => Err(WorkflowError::ExecutionError(format!(
911                "Sub-workflow execution timeout after {}ms",
912                timeout_ms
913            ))),
914        }
915    }

919    /// Execute a BayesianAggregate node - combines multiple claims into a single belief
920    async fn execute_bayesian_aggregate_node(
921        &self,
922        ctx: ExecutionContext,
923        node: &WorkflowNode,
924    ) -> Result<NodeResult> {
925        // Get claims from input variable
926        let claims_key = node
927            .config
928            .get("claims")
929            .and_then(|c| c.as_str())
930            .unwrap_or("claims");
931
932        let claims_array = ctx
933            .variables
934            .get(claims_key)
935            .or_else(|| ctx.inputs.get(claims_key))
936            .and_then(|v| v.as_array())
937            .cloned()
938            .ok_or_else(|| {
939                WorkflowError::ExecutionError(format!("Claims not found: {}", claims_key))
940            })?;
941
942        // Check if deduplication is enabled
943        let deduplicate = node
944            .config
945            .get("deduplicate")
946            .and_then(|d| d.as_bool())
947            .unwrap_or(true);
948
949        // Perform Bayesian aggregation
950        let aggregated_confidence = if deduplicate {
951            // Convert JSON claims to EpistemicClaim (simplified for workflow)
952            // In production, this would deserialize properly from the store
953            warn!(
954                "Bayesian aggregate with deduplication requires full EpistemicClaim deserialization"
955            );
956            0.5 // Placeholder - requires full model
957        } else {
958            // For now, extract confidence values from claims array
959            let confidences: Vec<f64> = claims_array
960                .iter()
961                .filter_map(|c| c.get("confidence").and_then(|v| v.as_f64()))
962                .collect();
963
964            if confidences.is_empty() {
965                return Err(WorkflowError::ExecutionError(
966                    "No valid confidence values found in claims".to_string(),
967                ));
968            }
969
970            // Simple average for now - full Bayesian would use log_odds
971            // This is a simplified implementation for workflow integration
972            let sum: f64 = confidences.iter().sum();
973            let avg = sum / confidences.len() as f64;
974
975            // Boost confidence when multiple sources agree
976            if confidences.len() > 1 {
977                (avg + 0.1).min(1.0)
978            } else {
979                avg
980            }
981        };
982
983        let output_key = node
984            .config
985            .get("output")
986            .and_then(|o| o.as_str())
987            .unwrap_or("aggregated_confidence");
988
989        info!(
990            node_id = %node.id,
991            claims_count = claims_array.len(),
992            aggregated_confidence = aggregated_confidence,
993            "Bayesian aggregation complete"
994        );
995
996        Ok(NodeResult {
997            node_id: node.id.clone(),
998            success: true,
999            output: serde_json::json!({
1000                output_key: aggregated_confidence,
1001                "sources_count": claims_array.len(),
1002                "deduplicated": deduplicate
1003            }),
1004            duration_ms: 0,
1005            error: None,
1006        })
1007    }

1107    /// Execute a ConflictDetection node - detects conflicts between claims
1108    async fn execute_conflict_detection_node(
1109        &self,
1110        ctx: ExecutionContext,
1111        node: &WorkflowNode,
1112    ) -> Result<NodeResult> {
1113        // Get new claim from input
1114        let new_claim_key = node
1115            .config
1116            .get("new_claim")
1117            .and_then(|c| c.as_str())
1118            .unwrap_or("new_claim");
1119
1120        let new_claim_json = ctx
1121            .variables
1122            .get(new_claim_key)
1123            .or_else(|| ctx.inputs.get(new_claim_key))
1124            .cloned()
1125            .ok_or_else(|| {
1126                WorkflowError::ExecutionError(format!("New claim not found: {}", new_claim_key))
1127            })?;
1128
1129        // Get existing claims to compare against
1130        let existing_claims_key = node
1131            .config
1132            .get("existing_claims")
1133            .and_then(|c| c.as_str())
1134            .unwrap_or("existing_claims");
1135
1136        let existing_claims = ctx
1137            .variables
1138            .get(existing_claims_key)
1139            .or_else(|| ctx.inputs.get(existing_claims_key))
1140            .and_then(|v| v.as_array())
1141            .cloned()
1142            .unwrap_or_default();
1143
1144        // For simplified workflow execution, we'll do content-based conflict detection
1145        // In production, this would use the full EpistemicClaim model
1146        let new_content = new_claim_json
1147            .get("content")
1148            .and_then(|c| c.as_str())
1149            .unwrap_or("");
1150
1151        let mut conflicts = Vec::new();
1152
1153        for existing in &existing_claims {
1154            let existing_content = existing
1155                .get("content")
1156                .and_then(|c| c.as_str())
1157                .unwrap_or("");
1158
1159            // Simple conflict detection: check for negation patterns
1160            let has_negation = |s: &str| -> bool {
1161                let lower = s.to_lowercase();
1162                lower.contains("not ") || lower.contains("no ") || lower.contains("never")
1163            };
1164
1165            if has_negation(new_content) != has_negation(existing_content) {
1166                // Check if they're about the same topic (simple word overlap)
1167                let new_words: std::collections::HashSet<_> =
1168                    new_content.split_whitespace().collect();
1169                let existing_words: std::collections::HashSet<_> =
1170                    existing_content.split_whitespace().collect();
1171                let overlap: std::collections::HashSet<_> =
1172                    new_words.intersection(&existing_words).collect();
1173
1174                if !overlap.is_empty() && overlap.len() >= 3 {
1175                    conflicts.push(serde_json::json!({
1176                        "existing_claim_id": existing.get("id").cloned().unwrap_or(Value::Null),
1177                        "conflict_type": "DirectContradiction",
1178                        "similarity": overlap.len() as f64 / new_words.len().max(existing_words.len()) as f64
1179                    }));
1180                }
1181            }
1182        }
1183
1184        let conflict_count = conflicts.len();
1185        let has_conflicts = conflict_count > 0;
1186
1187        info!(
1188            node_id = %node.id,
1189            conflict_count = conflict_count,
1190            "Conflict detection complete"
1191        );
1192
1193        let output_key = node
1194            .config
1195            .get("output")
1196            .and_then(|o| o.as_str())
1197            .unwrap_or("conflicts");
1198
1199        Ok(NodeResult {
1200            node_id: node.id.clone(),
1201            success: true,
1202            output: serde_json::json!({
1203                output_key: conflicts,
1204                "has_conflicts": has_conflicts,
1205                "conflict_count": conflict_count
1206            }),
1207            duration_ms: 0,
1208            error: None,
1209        })
1210    }
