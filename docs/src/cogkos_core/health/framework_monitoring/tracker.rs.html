1//! Framework Health Monitoring - Prediction accuracy tracking
2
3use chrono::{DateTime, Utc};
4
5/// Prediction accuracy tracking for insights
6#[derive(Debug, Clone)]
7pub struct PredictionTracker {
8    /// Insight ID
9    pub insight_id: uuid::Uuid,
10    /// Total predictions made
11    pub total_predictions: u64,
12    /// Correct predictions
13    pub correct_predictions: u64,
14    /// Prediction history
15    pub history: Vec<PredictionRecord>,
16    /// Accuracy trend
17    pub accuracy_trend: Vec<(DateTime<Utc>, f64)>,
18}
19
20#[derive(Debug, Clone)]
21pub struct PredictionRecord {
22    pub timestamp: DateTime<Utc>,
23    pub predicted_value: String,
24    pub actual_value: String,
25    pub was_correct: bool,
26    pub confidence: f64,
27    pub error_margin: f64,
28}
29
30impl PredictionTracker {
31    pub fn new(insight_id: uuid::Uuid) -> Self {
32        Self {
33            insight_id,
34            total_predictions: 0,
35            correct_predictions: 0,
36            history: Vec::new(),
37            accuracy_trend: Vec::new(),
38        }
39    }
40
41    /// Record a prediction outcome
42    pub fn record(&mut self, predicted: &str, actual: &str, confidence: f64) {
43        let was_correct = predicted == actual;
44        let error_margin = if was_correct { 0.0 } else { 1.0 };
45
46        self.total_predictions += 1;
47        if was_correct {
48            self.correct_predictions += 1;
49        }
50
51        self.history.push(PredictionRecord {
52            timestamp: Utc::now(),
53            predicted_value: predicted.to_string(),
54            actual_value: actual.to_string(),
55            was_correct,
56            confidence,
57            error_margin,
58        });
59
60        // Update trend every 10 predictions
61        if self.total_predictions.is_multiple_of(10) {
62            let accuracy = self.current_accuracy();
63            self.accuracy_trend.push((Utc::now(), accuracy));
64        }
65    }
66
67    /// Get current accuracy
68    pub fn current_accuracy(&self) -> f64 {
69        if self.total_predictions == 0 {
70            0.0
71        } else {
72            self.correct_predictions as f64 / self.total_predictions as f64
73        }
74    }
75
76    /// Get rolling accuracy over last N predictions
77    pub fn rolling_accuracy(&self, window: usize) -> f64 {
78        let window = window.min(self.history.len());
79        if window == 0 {
80            return 0.0;
81        }
82
83        let recent: Vec<_> = self.history.iter().rev().take(window).collect();
84        let correct = recent.iter().filter(|r| r.was_correct).count();
85
86        correct as f64 / window as f64
87    }
88
89    /// Check if accuracy is declining
90    pub fn is_declining(&self, window: usize) -> bool {
91        if self.accuracy_trend.len() < 2 {
92            return false;
93        }
94
95        let recent: Vec<_> = self.accuracy_trend.iter().rev().take(window).collect();
96        if recent.len() < 2 {
97            return false;
98        }
99
100        let first = recent[recent.len() - 1].1;
101        let last = recent[0].1;
102
103        last < first - 0.1 // Declined by more than 10%
104    }
105}
