8/// Insight extracted from conflict analysis
9#[derive(Debug, Clone)]
10pub struct ExtractedInsight {
11    pub id: uuid::Uuid,
12    pub content: String,
13    pub confidence: f64,
14    pub source_conflicts: Vec<uuid::Uuid>,
15    pub insight_type: InsightType,
16    pub supporting_claims: Vec<uuid::Uuid>,
17    pub key_entities: Vec<String>,
18    pub temporal_context: Option<TemporalContext>,
19}
20
21/// Type of insight extracted
22#[derive(Debug, Clone, Copy, PartialEq, Eq)]
23pub enum InsightType {
24    /// Different sources have different data
25    SourceDiscrepancy,
26    /// Conditions or context explain the difference
27    ContextualQualifier,
28    /// Change over time
29    TemporalEvolution,
30    /// Domain boundary or limitation
31    DomainBoundary,
32    /// Uncertainty or confidence issue
33    UncertaintyIndicator,
34    /// Emerging pattern
35    EmergingPattern,
36}
37
38/// Temporal context for insights
39#[derive(Debug, Clone)]
40pub struct TemporalContext {
41    pub valid_from: Option<chrono::DateTime<chrono::Utc>>,
42    pub valid_until: Option<chrono::DateTime<chrono::Utc>>,
43    pub trend_direction: TrendDirection,
44}
45
46#[derive(Debug, Clone, Copy, PartialEq, Eq)]
47pub enum TrendDirection {
48    Increasing,
49    Decreasing,
50    Stable,
51    Fluctuating,
52    Unknown,
53}
54
55/// Configuration for insight extraction
56#[derive(Clone, Debug)]
57pub struct InsightExtractionConfig {
58    /// Minimum number of conflicts to trigger analysis
59    pub min_conflicts: usize,
60    /// Minimum conflict density threshold
61    pub min_conflict_density: f64,
62    /// Confidence threshold for extracted insights
63    pub min_insight_confidence: f64,
64    /// Enable LLM-based insight extraction
65    pub use_llm_extraction: bool,
66}
67
68impl Default for InsightExtractionConfig {
69    fn default() -> Self {
70        Self {
71            min_conflicts: 3,
72            min_conflict_density: 0.1,
73            min_insight_confidence: 0.6,
74            use_llm_extraction: false,
75        }
76    }
77}
