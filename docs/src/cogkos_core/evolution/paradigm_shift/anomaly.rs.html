9/// Paradigm shift errors
10#[derive(Error, Debug)]
11pub enum ParadigmShiftError {
12    #[error("Anomaly detection failed: {0}")]
13    AnomalyDetectionFailed(String),
14    #[error("Sandbox error: {0}")]
15    SandboxError(String),
16    #[error("A/B test error: {0}")]
17    ABTestError(String),
18    #[error("Framework validation failed: {0}")]
19    ValidationFailed(String),
20    #[error("Rollback failed: {0}")]
21    RollbackFailed(String),
22    #[error("Insufficient data: {0}")]
23    InsufficientData(String),
24}
25
26pub type Result<T> = std::result::Result<T, ParadigmShiftError>;

28/// Anomaly detection result
29#[derive(Debug, Clone)]
30pub struct AnomalyDetectionResult {
31    /// Anomaly score (0.0 - 1.0), higher = more anomalous
32    pub anomaly_score: f64,
33    /// Detected anomalies
34    pub anomalies: Vec<Anomaly>,
35    /// Overall assessment
36    pub assessment: AnomalyAssessment,
37    /// Recommended action
38    pub recommendation: AnomalyRecommendation,
39}

41/// Individual anomaly
42#[derive(Debug, Clone)]
43pub struct Anomaly {
44    /// Type of anomaly
45    pub anomaly_type: AnomalyType,
46    /// Affected claims
47    pub claim_ids: Vec<Uuid>,
48    /// Description
49    pub description: String,
50    /// Confidence in detection
51    pub confidence: f64,
52    /// Deviation from expected pattern
53    pub deviation_score: f64,
54}

56/// Types of anomalies
57#[derive(Debug, Clone, Copy, PartialEq, Eq)]
58pub enum AnomalyType {
59    /// Conflicts that don't fit current resolution rules
60    ResolutionConflict,
61    /// Predictions consistently failing
62    PredictionFailure,
63    /// Knowledge structure doesn't fit current ontology
64    OntologyMismatch,
65    /// Confidence calibration is off
66    CalibrationDrift,
67    /// Semantic drift in key terms
68    SemanticDrift,
69    /// Coverage gap in critical domain
70    CoverageGap,
71}

73/// Overall anomaly assessment
74#[derive(Debug, Clone, Copy, PartialEq, Eq)]
75pub enum AnomalyAssessment {

77    Elevated,

80}

91/// Anomaly detector
92pub struct AnomalyDetector {
93    /// Threshold for considering an anomaly significant
94    _threshold: f64,
95    /// History of anomaly scores
96    history: Vec<(DateTime<Utc>, f64)>,
97    /// Pattern detector configuration
98    config: AnomalyConfig,
99}

101#[derive(Debug, Clone)]
102pub struct AnomalyConfig {
103    /// Prediction error threshold
104    pub prediction_error_threshold: f64,
105    /// Minimum sample size for detection
106    pub min_sample_size: usize,
107    /// Conflict density threshold
108    pub conflict_density_threshold: f64,
109    /// Semantic drift threshold
110    pub semantic_drift_threshold: f64,
111    /// Calibration drift threshold
112    pub calibration_drift_threshold: f64,
113}
114
115impl Default for AnomalyConfig {
116    fn default() -> Self {
117        Self {
118            prediction_error_threshold: 0.3,
119            min_sample_size: 100,
120            conflict_density_threshold: 0.2,
121            semantic_drift_threshold: 0.25,
122            calibration_drift_threshold: 0.15,
123        }
124    }
125}

127impl AnomalyDetector {
128    pub fn new(threshold: f64) -> Self {
129        Self {
130            _threshold: threshold,
131            history: Vec::new(),
132            config: AnomalyConfig::default(),
133        }
134    }
135
136    pub fn with_config(mut self, config: AnomalyConfig) -> Self {
137        self.config = config;
138        self
139    }

141    /// Detect anomalies in recent claims
142    pub fn detect(
143        &mut self,
144        claims: &[EpistemicClaim],
145        prediction_errors: &[f64],
146    ) -> AnomalyDetectionResult {
147        let mut anomalies = Vec::new();
148
149        // Check prediction failures
150        if prediction_errors.len() >= self.config.min_sample_size {
151            let avg_error: f64 =
152                prediction_errors.iter().sum::<f64>() / prediction_errors.len() as f64;
153            if avg_error > self.config.prediction_error_threshold {
154                anomalies.push(Anomaly {
155                    anomaly_type: AnomalyType::PredictionFailure,
156                    claim_ids: claims.iter().take(10).map(|c| c.id).collect(),
157                    description: format!(
158                        "Average prediction error {:.2} exceeds threshold",
159                        avg_error
160                    ),
161                    confidence: (avg_error / self.config.prediction_error_threshold).min(1.0),
162                    deviation_score: avg_error,
163                });
164            }
165        }
166
167        // Check conflict density
168        let contested_count = claims
169            .iter()
170            .filter(|c| {
171                matches!(
172                    c.epistemic_status,
173                    crate::models::EpistemicStatus::Contested
174                )
175            })
176            .count();
177        let conflict_density = if claims.is_empty() {
178            0.0
179        } else {
180            contested_count as f64 / claims.len() as f64
181        };
182
183        if conflict_density > self.config.conflict_density_threshold {
184            anomalies.push(Anomaly {
185                anomaly_type: AnomalyType::ResolutionConflict,
186                claim_ids: claims
187                    .iter()
188                    .filter(|c| {
189                        matches!(
190                            c.epistemic_status,
191                            crate::models::EpistemicStatus::Contested
192                        )
193                    })
194                    .take(10)
195                    .map(|c| c.id)
196                    .collect(),
197                description: format!("Conflict density {:.2} exceeds threshold", conflict_density),
198                confidence: (conflict_density / self.config.conflict_density_threshold).min(1.0),
199                deviation_score: conflict_density,
200            });
201        }
202
203        // Check confidence calibration
204        let high_confidence_wrong: Vec<_> = claims
205            .iter()
206            .filter(|c| {
207                c.confidence > 0.8 && c.last_prediction_error.map(|e| e > 0.5).unwrap_or(false)
208            })
209            .collect();
210
211        if high_confidence_wrong.len() as f64 / claims.len().max(1) as f64
212            > self.config.calibration_drift_threshold
213        {
214            anomalies.push(Anomaly {
215                anomaly_type: AnomalyType::CalibrationDrift,
216                claim_ids: high_confidence_wrong
217                    .iter()
218                    .take(10)
219                    .map(|c| c.id)
220                    .collect(),
221                description: "High confidence predictions frequently wrong".to_string(),
222                confidence: 0.8,
223                deviation_score: high_confidence_wrong.len() as f64 / claims.len() as f64,
224            });
225        }
226
227        // Calculate overall anomaly score
228        let anomaly_score = if anomalies.is_empty() {
229            0.0
230        } else {
231            let max_confidence: f64 = anomalies.iter().map(|a| a.confidence).fold(0.0, f64::max);
232            let avg_deviation: f64 =
233                anomalies.iter().map(|a| a.deviation_score).sum::<f64>() / anomalies.len() as f64;
234            (max_confidence * 0.6 + avg_deviation.min(1.0) * 0.4).min(1.0)
235        };
236
237        // Record in history
238        self.history.push((Utc::now(), anomaly_score));
239        if self.history.len() > 1000 {
240            self.history.remove(0);
241        }
242
243        // Determine assessment
244        let assessment = if anomaly_score >= 0.8 {
245            AnomalyAssessment::ParadigmBreaking
246        } else if anomaly_score > 0.6 {
247            AnomalyAssessment::Critical
248        } else if anomaly_score > 0.3 {
249            AnomalyAssessment::Elevated
250        } else {
251            AnomalyAssessment::Normal
252        };
253
254        // Determine recommendation
255        let recommendation = match assessment {
256            AnomalyAssessment::ParadigmBreaking => AnomalyRecommendation::InitiateParadigmShift,
257            AnomalyAssessment::Critical => AnomalyRecommendation::ReviewFramework,
258            AnomalyAssessment::Elevated => AnomalyRecommendation::Monitor,
259            AnomalyAssessment::Normal => AnomalyRecommendation::NoAction,
260        };
261
262        AnomalyDetectionResult {
263            anomaly_score,
264            anomalies,
265            assessment,
266            recommendation,
267        }
268    }

270    /// Check if there's a persistent anomaly trend
271    pub fn has_persistent_anomaly(&self, window_size: usize, threshold: f64) -> bool {
272        if self.history.len() < window_size {
273            return false;
274        }
275
276        let recent: Vec<f64> = self
277            .history
278            .iter()
279            .rev()
280            .take(window_size)
281            .map(|(_, s)| *s)
282            .collect();
283        let avg = recent.iter().sum::<f64>() / recent.len() as f64;
284
285        avg > threshold
286    }
287
288    /// Get anomaly trend (positive = worsening)
289    pub fn trend(&self, window_size: usize) -> f64 {
290        if self.history.len() < window_size * 2 {
291            return 0.0;
292        }
293
294        let recent: Vec<f64> = self
295            .history
296            .iter()
297            .rev()
298            .take(window_size)
299            .map(|(_, s)| *s)
300            .collect();
301        let older: Vec<f64> = self
302            .history
303            .iter()
304            .rev()
305            .skip(window_size)
306            .take(window_size)
307            .map(|(_, s)| *s)
308            .collect();
309
310        let recent_avg = recent.iter().sum::<f64>() / recent.len() as f64;
311        let older_avg = older.iter().sum::<f64>() / older.len() as f64;
312
313        recent_avg - older_avg
314    }
