1/// Calculate decayed confidence over time
2/// Uses exponential decay modulated by activation weight
3///
4/// Formula: new_confidence = confidence * exp(-lambda * time_delta / activation_weight)
5///
6/// Higher activation_weight slows decay (knowledge that is accessed frequently persists longer)
7pub fn calculate_decay(
8    confidence: f64,
9    lambda: f64, // Base decay rate (e.g., 0.01 for slow decay)
10    time_delta_hours: f64,
11    activation_weight: f64,
12) -> f64 {
13    if confidence <= 0.0 {
14        return 0.0;
15    }
16
17    // Activation weight protects against decay (0.5 = normal, 1.0 = max protection)
18    let protection_factor = activation_weight.max(0.1); // Min 0.1 to prevent division by zero
19    let effective_lambda = lambda / protection_factor;
20
21    let decay_factor = (-effective_lambda * time_delta_hours).exp();
22    let new_confidence = confidence * decay_factor;
23
24    new_confidence.clamp(0.0, 1.0)
25}

27/// Calculate decay with revalidation boost
28/// If the knowledge is revalidated (e.g., by new confirming evidence),
29/// the confidence gets a boost
30pub fn calculate_decay_with_revalidation(
31    confidence: f64,
32    lambda: f64,
33    time_delta_hours: f64,
34    activation_weight: f64,
35    revalidation_boost: f64, // 0.0 to 1.0 boost from confirmation
36) -> f64 {
37    let decayed = calculate_decay(confidence, lambda, time_delta_hours, activation_weight);
38
39    // Apply revalidation boost
40    let boosted = decayed + revalidation_boost * (1.0 - decayed);
41    boosted.clamp(0.0, 1.0)
42}

44/// Check if knowledge needs revalidation
45pub fn needs_revalidation(
46    current_confidence: f64,
47    threshold: f64,
48    last_validated_hours: f64,
49    max_age_hours: f64,
50) -> bool {
51    current_confidence < threshold || last_validated_hours > max_age_hours
52}

54/// Calculate effective durability based on usage patterns
55pub fn calculate_effective_durability(
56    base_durability: f64,
57    access_count: u64,
58    confirmation_count: u64,
59) -> f64 {
60    let usage_bonus = (access_count as f64 * 0.01).min(0.2); // Max 0.2 bonus from usage
61    let confirmation_bonus = (confirmation_count as f64 * 0.02).min(0.3); // Max 0.3 from confirmation
62
63    (base_durability + usage_bonus + confirmation_bonus).min(1.0)
64}
