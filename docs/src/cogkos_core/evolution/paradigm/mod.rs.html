1//! Paradigm Shift Module - L2 Evolution Engine (Paradigm Shift Mode)
2//!
3//! This module implements the paradigm shift functionality for CogKOS:

20/// Configuration for anomaly detection
21#[derive(Debug, Clone, Serialize, Deserialize)]
22pub struct AnomalyConfig {
23    /// Prediction error streak threshold
24    pub prediction_error_streak_threshold: u32,
25    /// Conflict density threshold (0.0 - 1.0)
26    pub conflict_density_threshold: f64,
27    /// Cache hit rate decline threshold (negative percentage)
28    pub cache_hit_rate_decline_threshold: f64,
29    /// Minimum samples before triggering
30    pub min_samples: usize,
31    /// Time window for analysis (hours)
32    pub analysis_window_hours: i64,
33    /// Consecutive anomaly periods required
34    pub consecutive_periods_required: u32,
35}
36
37impl Default for AnomalyConfig {
38    fn default() -> Self {
39        Self {
40            prediction_error_streak_threshold: 5,
41            conflict_density_threshold: 0.3,
42            cache_hit_rate_decline_threshold: -0.2,
43            min_samples: 100,
44            analysis_window_hours: 24,
45            consecutive_periods_required: 3,
46        }
47    }
48}

63/// Anomaly detection result
64#[derive(Debug, Clone, Serialize, Deserialize)]
65pub struct AnomalyResult {
66    pub is_anomaly: bool,
67    pub signals: Vec<AnomalySignal>,
68    pub severity: f64,

72#[derive(Debug, Clone, Serialize, Deserialize)]
73pub enum AnomalySignal {
74    HighPredictionErrorStreak { streak: u32, threshold: u32 },
75    ElevatedConflictDensity { density: f64, threshold: f64 },
76    DecliningCacheHitRate { trend: f64, threshold: f64 },
77    LowInsightAccuracy { accuracy: f64, expected: f64 },
78    SystematicBiasDetected { bias_type: String, magnitude: f64 },
79}

81#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
82pub enum ShiftRecommendation {
