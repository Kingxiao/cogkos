9use std::sync::{Arc, RwLock};

12/// Standard Prometheus histogram bucket boundaries
13pub const HISTOGRAM_BUCKETS: &[f64] = &[
14    0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0,
15];
16

21    tracing::info!("Logging initialized");
22}

24/// Metrics collector for Prometheus export
25#[derive(Clone)]
26pub struct MetricsCollector {
27    counters: Arc<RwLock<HashMap<String, u64>>>,
28    gauges: Arc<RwLock<HashMap<String, f64>>>,
29    histograms: Arc<RwLock<HashMap<String, Vec<f64>>>>,
30}
31
32impl MetricsCollector {
33    pub fn new() -> Self {
34        Self {
35            counters: Arc::new(RwLock::new(HashMap::new())),
36            gauges: Arc::new(RwLock::new(HashMap::new())),
37            histograms: Arc::new(RwLock::new(HashMap::new())),
38        }
39    }
40
41    /// Increment a counter
42    pub fn inc_counter(&self, name: &str, value: u64) {
43        if let Ok(mut counters) = self.counters.write() {
44            *counters.entry(name.to_string()).or_insert(0) += value;
45        }
46    }
47
48    /// Set a gauge value
49    pub fn set_gauge(&self, name: &str, value: f64) {
50        if let Ok(mut gauges) = self.gauges.write() {
51            gauges.insert(name.to_string(), value);
52        }
53    }
54
55    /// Observe a value for a histogram metric
56    pub fn observe_histogram(&self, name: &str, value: f64) {
57        if let Ok(mut histograms) = self.histograms.write() {
58            histograms
59                .entry(name.to_string())
60                .or_insert_with(Vec::new)
61                .push(value);
62        }
63    }
64
65    /// Record a duration as seconds into a histogram
66    pub fn record_duration(&self, name: &str, duration: Duration) {
67        self.observe_histogram(name, duration.as_secs_f64());
68    }
69
70    /// Generate Prometheus metrics text
71    pub fn to_prometheus_text(&self) -> String {
72        let mut output = String::new();
73
74        if let Ok(counters) = self.counters.read() {
75            for (name, value) in counters.iter() {
76                let safe_name = name.replace('-', "_");
77                output.push_str(&format!(
78                    "# TYPE {} counter\n{} {}\n",
79                    safe_name, safe_name, value
80                ));
81            }
82        }
83
84        if let Ok(gauges) = self.gauges.read() {
85            for (name, value) in gauges.iter() {
86                let safe_name = name.replace('-', "_");
87                output.push_str(&format!(
88                    "# TYPE {} gauge\n{} {}\n",
89                    safe_name, safe_name, value
90                ));
91            }
92        }
93
94        if let Ok(histograms) = self.histograms.read() {
95            for (name, values) in histograms.iter() {
96                let safe_name = name.replace('-', "_");
97                let count = values.len();
98                let sum: f64 = values.iter().sum();
99
100                output.push_str(&format!("# TYPE {} histogram\n", safe_name));
101
102                for bucket in HISTOGRAM_BUCKETS {
103                    let bucket_count = values.iter().filter(|&&v| v <= *bucket).count();
104                    output.push_str(&format!(
105                        "{}_bucket{{le=\"{}\"}} {}\n",
106                        safe_name, bucket, bucket_count
107                    ));
108                }
109                // +Inf bucket always equals total count
110                output.push_str(&format!("{}_bucket{{le=\"+Inf\"}} {}\n", safe_name, count));
111                output.push_str(&format!("{}_sum {}\n", safe_name, sum));
112                output.push_str(&format!("{}_count {}\n", safe_name, count));
113            }
114        }
115
116        output
117    }
118}
119
120impl Default for MetricsCollector {
121    fn default() -> Self {
122        Self::new()
123    }
124}

139    #[test]
140    fn test_counter_increment() {

142        collector.inc_counter("test_counter", 1);

158    #[test]
159    fn test_observe_histogram() {
160        let collector = MetricsCollector::new();
161        collector.observe_histogram("request_duration", 0.003);
162        collector.observe_histogram("request_duration", 0.07);
163        collector.observe_histogram("request_duration", 0.5);
164        collector.observe_histogram("request_duration", 3.0);
165
166        let text = collector.to_prometheus_text();
167
168        assert!(text.contains("# TYPE request_duration histogram"));
169        // 0.003 <= 0.005, so bucket le=0.005 should be 1
170        assert!(text.contains("request_duration_bucket{le=\"0.005\"} 1"));
171        // 0.003, 0.07 <= 0.1, so bucket le=0.1 should be 2
172        assert!(text.contains("request_duration_bucket{le=\"0.1\"} 2"));
173        // 0.003, 0.07, 0.5 <= 0.5, so bucket le=0.5 should be 3
174        assert!(text.contains("request_duration_bucket{le=\"0.5\"} 3"));
175        // 0.003, 0.07, 0.5, 3.0 <= 5.0, so bucket le=5 should be 4
176        assert!(text.contains("request_duration_bucket{le=\"5\"} 4"));
177        // +Inf always equals count
178        assert!(text.contains("request_duration_bucket{le=\"+Inf\"} 4"));
179        assert!(text.contains("request_duration_count 4"));
180        // sum = 0.003 + 0.07 + 0.5 + 3.0 = 3.573
181        assert!(text.contains("request_duration_sum 3.573"));
182    }
