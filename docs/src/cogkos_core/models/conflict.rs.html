1use serde::{Deserialize, Serialize};

4/// Conflict record between two claims
5#[derive(Debug, Clone, Serialize, Deserialize)]
6pub struct ConflictRecord {
7    pub id: Uuid,
8    pub tenant_id: String,
9    pub claim_a_id: Uuid,
10    pub claim_b_id: Uuid,
11    pub conflict_type: ConflictType,
12    pub severity: f64,
13    #[serde(skip_serializing_if = "Option::is_none")]
14    pub description: Option<String>,
15    pub detected_at: chrono::DateTime<chrono::Utc>,
16    #[serde(skip_serializing_if = "Option::is_none")]
17    pub resolved_at: Option<chrono::DateTime<chrono::Utc>>,
18    pub resolution_status: ResolutionStatus,
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub resolution: Option<serde_json::Value>,
21    #[serde(skip_serializing_if = "Option::is_none")]
22    pub resolution_note: Option<String>,
23    #[serde(skip_serializing_if = "Option::is_none")]
24    pub elevated_insight_id: Option<Uuid>,
25}
26
27impl ConflictRecord {
28    /// Create a new conflict record
29    pub fn new(
30        tenant_id: impl Into<String>,
31        claim_a_id: Uuid,
32        claim_b_id: Uuid,
33        conflict_type: ConflictType,
34    ) -> Self {
35        let desc = format!(
36            "{:?} between {} and {}",
37            conflict_type, claim_a_id, claim_b_id
38        );
39        Self {
40            id: Uuid::new_v4(),
41            tenant_id: tenant_id.into(),
42            claim_a_id,
43            claim_b_id,
44            conflict_type,
45            severity: 0.5,
46            description: Some(desc),
47            detected_at: chrono::Utc::now(),
48            resolved_at: None,
49            resolution_status: ResolutionStatus::Open,
50            resolution: None,
51            resolution_note: None,
52            elevated_insight_id: None,
53        }
54    }
55
56    /// Mark as elevated to insight
57    pub fn elevate(&mut self, insight_id: Uuid) {
58        self.resolution_status = ResolutionStatus::Elevated;
59        self.elevated_insight_id = Some(insight_id);
60    }
61
62    /// Mark as dismissed
63    pub fn dismiss(&mut self, note: impl Into<String>) {
64        self.resolution_status = ResolutionStatus::Dismissed;
65        self.resolution_note = Some(note.into());
66    }
67
68    /// Mark as accepted
69    pub fn accept(&mut self, note: impl Into<String>) {
70        self.resolution_status = ResolutionStatus::Accepted;
71        self.resolution_note = Some(note.into());
72    }
73}

75/// Type of conflict
76#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
77#[serde(rename_all = "PascalCase")]
78pub enum ConflictType {
79    DirectContradiction,
80    ContextDependent,
81    TemporalShift,
82    TemporalInconsistency,
83    SourceDisagreement,
84    ConfidenceMismatch,
85    ContextualDifference,
86}
87
88impl ConflictType {
89    pub fn as_db_str(&self) -> &'static str {
90        match self {
91            ConflictType::DirectContradiction => "direct_contradiction",
92            ConflictType::ContextDependent => "context_dependent",
93            ConflictType::TemporalShift => "temporal_shift",
94            ConflictType::TemporalInconsistency => "temporal_inconsistency",
95            ConflictType::SourceDisagreement => "source_disagreement",
96            ConflictType::ConfidenceMismatch => "confidence_mismatch",
97            ConflictType::ContextualDifference => "contextual_difference",
98        }
99    }
100
101    pub fn from_db_str(s: &str) -> Self {
102        match s.to_lowercase().as_str() {
103            "direct_contradiction" | "directcontradiction" => ConflictType::DirectContradiction,
104            "context_dependent" | "contextdependent" => ConflictType::ContextDependent,
105            "temporal_shift" | "temporalshift" => ConflictType::TemporalShift,
106            "temporal_inconsistency" | "temporalinconsistency" => {
107                ConflictType::TemporalInconsistency
108            }
109            "source_disagreement" | "sourcedisagreement" => ConflictType::SourceDisagreement,
110            "confidence_mismatch" | "confidencemismatch" => ConflictType::ConfidenceMismatch,
111            "contextual_difference" | "contextualdifference" => ConflictType::ContextualDifference,
112            _ => ConflictType::DirectContradiction,
113        }
114    }
115}
