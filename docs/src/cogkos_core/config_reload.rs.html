1//! Configuration hot reload module
2//!
3//! Provides:
4//! - File-based configuration watching
5//! - Runtime configuration updates
6//! - Configuration change callbacks
7
8use notify::{RecursiveMode, Watcher};
9use parking_lot::RwLock;
10use std::collections::HashMap;
11use std::path::Path;
12use std::sync::Arc;
13use tokio::sync::mpsc;
14
15/// Configuration change event
16#[derive(Clone, Debug)]
17pub enum ConfigEvent {
18    /// Configuration was modified
19    Modified(String),
20    /// Configuration was created
21    Created(String),
22    /// Configuration was removed
23    Removed(String),
24    /// Error occurred
25    Error(String),
26}
27
28/// Configuration change callback
29pub type ConfigCallback = Arc<dyn Fn(ConfigEvent) + Send + Sync>;
30
31/// Hot reload configuration manager
32pub struct ConfigHotReload {
33    watchers: Arc<RwLock<HashMap<String, notify::RecommendedWatcher>>>,
34    callbacks: Arc<RwLock<Vec<ConfigCallback>>>,
35    event_sender: Option<mpsc::Sender<ConfigEvent>>,
36}
37
38impl ConfigHotReload {
39    /// Create a new hot reload manager
40    pub fn new() -> Self {
41        Self {
42            watchers: Arc::new(RwLock::new(HashMap::new())),
43            callbacks: Arc::new(RwLock::new(Vec::new())),
44            event_sender: None,
45        }
46    }
47
48    /// Create with event channel
49    pub fn with_channel(channel_size: usize) -> (Self, mpsc::Receiver<ConfigEvent>) {
50        let (sender, receiver) = mpsc::channel(channel_size);
51        let mut this = Self::new();
52        this.event_sender = Some(sender);
53        (this, receiver)
54    }
55
56    /// Watch a configuration file
57    pub fn watch_file(&self, path: impl Into<String>) -> Result<(), String> {
58        let path = path.into();
59        let path_clone = path.clone();
60        let callbacks = self.callbacks.clone();
61        let sender = self.event_sender.clone();
62
63        let watcher_result =
64            notify::recommended_watcher(move |res: Result<notify::Event, notify::Error>| {
65                match res {
66                    Ok(event) => {
67                        for path in event.paths {
68                            let path_str = path.to_string_lossy().to_string();
69                            let config_event = match event.kind {
70                                notify::EventKind::Create(_) => {
71                                    ConfigEvent::Created(path_str.clone())
72                                }
73                                notify::EventKind::Modify(_) => {
74                                    ConfigEvent::Modified(path_str.clone())
75                                }
76                                notify::EventKind::Remove(_) => {
77                                    ConfigEvent::Removed(path_str.clone())
78                                }
79                                _ => continue,
80                            };
81
82                            // Notify callbacks
83                            for callback in callbacks.read().iter() {
84                                callback(config_event.clone());
85                            }
86
87                            // Send to channel if available
88                            if let Some(ref sender) = sender {
89                                let _ = sender.blocking_send(config_event);
90                            }
91                        }
92                    }
93                    Err(e) => {
94                        let config_event = ConfigEvent::Error(e.to_string());
95                        for callback in callbacks.read().iter() {
96                            callback(config_event.clone());
97                        }
98                        if let Some(ref sender) = sender {
99                            let _ = sender.blocking_send(config_event);
100                        }
101                    }
102                }
103            });
104
105        let mut watcher = watcher_result.map_err(|e| e.to_string())?;
106
107        // Watch the file
108        watcher
109            .watch(Path::new(&path), RecursiveMode::NonRecursive)
110            .map_err(|e| e.to_string())?;
111
112        self.watchers.write().insert(path_clone, watcher);
113
114        Ok(())
115    }
116
117    /// Watch a directory for config files
118    pub fn watch_directory(&self, dir: impl Into<String>) -> Result<(), String> {
119        let dir = dir.into();
120        let dir_clone = dir.clone();
121        let callbacks = self.callbacks.clone();
122        let sender = self.event_sender.clone();
123
124        let watcher_result =
125            notify::recommended_watcher(move |res: Result<notify::Event, notify::Error>| {
126                match res {
127                    Ok(event) => {
128                        for path in event.paths {
129                            let path_str = path.to_string_lossy().to_string();
130                            // Only process config files
131                            if !path_str.ends_with(".toml")
132                                && !path_str.ends_with(".json")
133                                && !path_str.ends_with(".yaml")
134                                && !path_str.ends_with(".yml")
135                            {
136                                continue;
137                            }
138
139                            let config_event = match event.kind {
140                                notify::EventKind::Create(_) => {
141                                    ConfigEvent::Created(path_str.clone())
142                                }
143                                notify::EventKind::Modify(_) => {
144                                    ConfigEvent::Modified(path_str.clone())
145                                }
146                                notify::EventKind::Remove(_) => {
147                                    ConfigEvent::Removed(path_str.clone())
148                                }
149                                _ => continue,
150                            };
151
152                            for callback in callbacks.read().iter() {
153                                callback(config_event.clone());
154                            }
155
156                            if let Some(ref sender) = sender {
157                                let _ = sender.blocking_send(config_event);
158                            }
159                        }
160                    }
161                    Err(e) => {
162                        let config_event = ConfigEvent::Error(e.to_string());
163                        for callback in callbacks.read().iter() {
164                            callback(config_event.clone());
165                        }
166                        if let Some(ref sender) = sender {
167                            let _ = sender.blocking_send(config_event);
168                        }
169                    }
170                }
171            });
172
173        let mut watcher = watcher_result.map_err(|e| e.to_string())?;
174        watcher
175            .watch(Path::new(&dir), RecursiveMode::Recursive)
176            .map_err(|e| e.to_string())?;
177
178        self.watchers.write().insert(dir_clone, watcher);
179
180        Ok(())
181    }
182
183    /// Register a callback for configuration changes
184    pub fn on_change(&self, callback: ConfigCallback) {
185        self.callbacks.write().push(callback);
186    }
187
188    /// Stop watching all files
189    pub fn stop(&self) {
190        self.watchers.write().clear();
191    }
192}

200/// Reloadable configuration wrapper
201pub struct ReloadableConfig<T> {
202    config: Arc<RwLock<T>>,
203    hot_reload: ConfigHotReload,
204}
205
206impl<T: Clone + serde::de::DeserializeOwned + Send + Sync + 'static> ReloadableConfig<T> {
207    /// Create a new reloadable config
208    pub fn new(path: &str) -> Result<Self, String> {
209        // Load initial config
210        let config = Self::load_config(path)?;
211
212        let this = Self {
213            config: Arc::new(RwLock::new(config)),
214            hot_reload: ConfigHotReload::new(),
215        };
216
217        // Watch for changes
218        let config_path = path.to_string();
219        let config_clone = this.config.clone();
220        this.hot_reload.on_change(Arc::new(move |event| {
221            if let ConfigEvent::Modified(ref event_path) = event
222                && event_path == &config_path
223            {
224                tracing::info!("Configuration file changed, reloading...");
225                match Self::load_config(event_path) {
226                    Ok(new_config) => {
227                        *config_clone.write() = new_config;
228                        tracing::info!("Configuration reloaded successfully");
229                    }
230                    Err(e) => {
231                        tracing::error!("Failed to reload configuration: {}", e);
232                    }
233                }
234            }
235        }));
236
237        this.hot_reload.watch_file(path)?;
238
239        Ok(this)
240    }
241
242    /// Load configuration from file
243    fn load_config(path: &str) -> Result<T, String> {
244        let content = std::fs::read_to_string(path).map_err(|e| e.to_string())?;
245
246        if path.ends_with(".toml") {
247            toml::from_str(&content).map_err(|e| e.to_string())
248        } else if path.ends_with(".json") {
249            serde_json::from_str(&content).map_err(|e| e.to_string())
250        } else if path.ends_with(".yaml") || path.ends_with(".yml") {
251            serde_yaml::from_str(&content).map_err(|e| e.to_string())
252        } else {
253            Err("Unsupported config file format".to_string())
254        }
255    }
256
257    /// Get current configuration
258    pub fn get(&self) -> T {
259        self.config.read().clone()
260    }
261
262    /// Get configuration reference
263    pub fn get_ref(&self) -> parking_lot::RwLockReadGuard<'_, T> {
264        self.config.read()
265    }
266}
