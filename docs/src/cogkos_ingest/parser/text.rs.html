3use crate::{DocumentParser, TextChunk};
4use async_trait::async_trait;
5use cogkos_core::{CogKosError, Result};
6
7/// Plain text parser
8pub struct TextParser;
9
10impl TextParser {
11    pub fn new() -> Self {
12        Self
13    }
14}
15
16impl Default for TextParser {
17    fn default() -> Self {
18        Self::new()
19    }
20}
21
22#[async_trait]
23impl DocumentParser for TextParser {
24    fn supported_extensions(&self) -> &[&str] {
25        &["txt"]
26    }
27
28    async fn parse(&self, data: &[u8], _filename: &str) -> Result<Vec<TextChunk>> {
29        let content = String::from_utf8(data.to_vec())
30            .map_err(|_| CogKosError::Parse("Invalid UTF-8 in text file".to_string()))?;
31
32        let cleaned = super::clean_text(&content);
33        let mut chunks = Vec::new();
34        let mut current_chunk = String::new();
35        let mut chunk_index = 0;
36        let max_chunk_size = 2000;
37
38        // Split into chunks by paragraphs
39        for paragraph in cleaned.split("\n\n") {
40            let para = paragraph.trim();
41            if para.is_empty() {
42                continue;
43            }
44
45            if current_chunk.len() + para.len() > max_chunk_size && !current_chunk.is_empty() {
46                chunks.push(TextChunk {
47                    content: current_chunk.trim().to_string(),
48                    chunk_index,
49                    metadata: std::collections::HashMap::new(),
50                });
51                chunk_index += 1;
52                current_chunk.clear();
53            }
54
55            current_chunk.push_str(para);
56            current_chunk.push('\n');
57        }
58
59        // Add remaining text
60        if !current_chunk.trim().is_empty() {
61            chunks.push(TextChunk {
62                content: current_chunk.trim().to_string(),
63                chunk_index,
64                metadata: std::collections::HashMap::new(),
65            });
66        }
67
68        // If no chunks were created
69        if chunks.is_empty() && !cleaned.trim().is_empty() {
70            chunks.push(TextChunk {
71                content: cleaned.trim().to_string(),
72                chunk_index: 0,
73                metadata: std::collections::HashMap::new(),
74            });
75        }
76
77        Ok(chunks)
78    }
79}
