25/// Parser registry for managing multiple parsers
26pub struct ParserRegistry {
27    parsers: Vec<Box<dyn DocumentParser>>,
28}
29
30impl ParserRegistry {
31    /// Create new registry with default parsers
32    pub fn new() -> Self {
33        let mut registry = Self {
34            parsers: Vec::new(),
35        };
36        registry.register(Box::new(MarkdownParser::new()));
37        registry.register(Box::new(PdfParser::new()));
38        registry.register(Box::new(DocxParser::new()));
39        registry.register(Box::new(ExcelParser::new()));
40        registry.register(Box::new(PptxParser::new()));
41        registry.register(Box::new(TextParser::new()));
42        registry.register(Box::new(ImageParser::default()));
43        registry.register(Box::new(AudioParser::default()));
44        registry.register(Box::new(VideoParser::default()));
45        registry
46    }
47
48    /// Register a parser
49    pub fn register(&mut self, parser: Box<dyn DocumentParser>) {
50        self.parsers.push(parser);
51    }
52
53    /// Get parser for filename
54    pub fn get_parser(&self, filename: &str) -> Option<&dyn DocumentParser> {
55        let extension = filename.split('.').next_back()?.to_lowercase();
56        self.parsers
57            .iter()
58            .find(|p| p.supported_extensions().contains(&extension.as_str()))
59            .map(|p| p.as_ref())
60    }
61
62    /// Parse file using appropriate parser
63    pub async fn parse(&self, data: &[u8], filename: &str) -> Result<Vec<TextChunk>> {
64        match self.get_parser(filename) {
65            Some(parser) => parser.parse(data, filename).await,
66            None => Err(CogKosError::Parse(format!(
67                "No parser found for file: {}",
68                filename
69            ))),
70        }
71    }
72}
73
74impl Default for ParserRegistry {
75    fn default() -> Self {
76        Self::new()
77    }
78}

80/// Chunk text into smaller pieces
81pub fn chunk_text(text: &str, max_chunk_size: usize, overlap: usize) -> Vec<String> {
82    if text.len() <= max_chunk_size {
83        return vec![text.to_string()];
84    }
85
86    let mut chunks = Vec::new();
87    let mut start = 0;
88
89    while start < text.len() {
90        let end = (start + max_chunk_size).min(text.len());
91        let chunk = &text[start..end];
92        chunks.push(chunk.to_string());
93
94        // Move start forward with overlap
95        start += max_chunk_size - overlap;
96
97        // Avoid infinite loop on very small progress
98        if start >= end {
99            break;
100        }
101    }
102
103    chunks
104}

106/// Clean extracted text
107pub fn clean_text(text: &str) -> String {
108    text.replace("\r\n", "\n")
109        .replace("\r", "\n")
110        .split('\n')
111        .map(|line| line.trim())
112        .filter(|line| !line.is_empty())
113        .collect::<Vec<_>>()
114        .join("\n")
115}
