68impl VideoParser {
69    pub fn new(config: VideoParserConfig) -> Self {

72            config,

76    /// Parse video file and extract audio for transcription
77    ///
78    /// # Arguments
79    /// * `video_data` - Raw video file bytes
80    /// * `format` - Video format (mp4, mov, avi, etc.)
81    ///
82    /// # Returns
83    /// VideoParseResult with audio transcription
84    pub async fn parse(
85        &self,
86        video_data: &[u8],
87        format: &str,
88    ) -> Result<VideoParseResult, VideoParserError> {
89        // Validate format
90        let supported_formats = ["mp4", "mov", "avi", "mkv", "webm"];
91        if !supported_formats.contains(&format.to_lowercase().as_str()) {
92            return Err(VideoParserError::UnsupportedFormat(format.to_string()));
93        }
94
95        // Extract video metadata (basic parsing for common formats)
96        let (duration, width, height, fps) = self.parse_video_metadata(video_data, format);
97
98        // Extract audio if enabled
99        let audio_transcription = if self.config.extract_audio {
100            match self.extract_audio_from_video(video_data, format).await {
101                Ok(audio_data) => {
102                    // Use AudioParser to transcribe
103                    Some(
104                        self.audio_parser
105                            .parse(&audio_data, "wav")
106                            .await
107                            .unwrap_or_else(|e| {
108                                tracing::warn!("Audio transcription failed: {}", e);
109                                crate::parser::AudioTranscription {
110                                    text: "[Audio extraction failed]".to_string(),
111                                    language: "unknown".to_string(),
112                                    duration_secs: duration,
113                                    segments: vec![],
114                                    confidence: 0.0,
115                                }
116                            }),
117                    )
118                }
119                Err(e) => {
120                    tracing::warn!("Failed to extract audio: {}", e);
121                    None
122                }
123            }
124        } else {
125            None
126        };
127
128        // Extract keyframes if enabled (placeholder - would need ffmpeg/opencv)
129        let keyframes: Vec<KeyFrame> = vec![];
130
131        Ok(VideoParseResult {
132            duration_secs: duration,
133            width,
134            height,
135            fps,
136            audio_transcription,
137            keyframes,
138        })
139    }

141    /// Parse basic video metadata from container
142    fn parse_video_metadata(&self, video_data: &[u8], format: &str) -> (f64, u32, u32, f64) {
143        // Basic MP4/MOV parsing to extract duration and dimensions
144        // This is a simplified implementation
145        match format.to_lowercase().as_str() {
146            "mp4" | "mov" => self.parse_mp4_metadata(video_data),
147            "avi" => (0.0, 0, 0, 0.0), // AVI parsing would require more complex implementation
148            "mkv" | "webm" => (0.0, 0, 0, 0.0),
149            _ => (0.0, 0, 0, 0.0),
150        }
151    }
152
153    /// Parse MP4/MOV container metadata
154    fn parse_mp4_metadata(&self, data: &[u8]) -> (f64, u32, u32, f64) {
155        // Very basic MP4 parsing - look for common atoms
156        // In production, use a proper MP4 parser
157
158        // Estimate duration based on file size (rough approximation)
159        // Assume ~1MB per second for HD video
160        let estimated_duration = data.len() as f64 / 1_000_000.0;
161
162        // Default dimensions (would be parsed from actual atoms)
163        let (width, height) = (1920, 1080); // Default to HD
164        let fps = 30.0;
165
166        (estimated_duration, width, height, fps)
167    }

169    /// Extract audio from video data using ffmpeg subprocess
170    async fn extract_audio_from_video(
171        &self,
172        video_data: &[u8],
173        format: &str,
174    ) -> Result<Vec<u8>, VideoParserError> {
175        // Check ffmpeg availability
176        let ffmpeg_check = tokio::process::Command::new("ffmpeg")
177            .arg("-version")
178            .output()
179            .await;
180        if ffmpeg_check.is_err() {
181            return Err(VideoParserError::FfmpegError(
182                "ffmpeg not found in PATH".to_string(),
183            ));
184        }
185
186        // Write video data to a temp file (ffmpeg needs seekable input for most containers)
187        let tmp_dir = std::env::temp_dir();
188        let input_id = uuid::Uuid::new_v4();
189        let input_path = tmp_dir.join(format!("cogkos_video_{}.{}", input_id, format));
190        let output_path = tmp_dir.join(format!("cogkos_audio_{}.wav", input_id));
191
192        tokio::fs::write(&input_path, video_data)
193            .await
194            .map_err(|e| VideoParserError::IoError(format!("Failed to write temp video: {}", e)))?;
195
196        // Run ffmpeg: extract audio → 16kHz mono WAV (Whisper-compatible)
197        let result = tokio::process::Command::new("ffmpeg")
198            .args([
199                "-i",
200                input_path.to_str().unwrap_or("input"),
201                "-vn", // no video
202                "-acodec",
203                "pcm_s16le", // 16-bit PCM
204                "-ar",
205                "16000", // 16kHz sample rate
206                "-ac",
207                "1",  // mono
208                "-y", // overwrite
209                output_path.to_str().unwrap_or("output"),
210            ])
211            .output()
212            .await
213            .map_err(|e| {
214                VideoParserError::FfmpegError(format!("ffmpeg execution failed: {}", e))
215            })?;
216
217        // Clean up input immediately
218        let _ = tokio::fs::remove_file(&input_path).await;
219
220        if !result.status.success() {
221            let _ = tokio::fs::remove_file(&output_path).await;
222            let stderr = String::from_utf8_lossy(&result.stderr);
223            return Err(VideoParserError::FfmpegError(format!(
224                "ffmpeg exited with {}: {}",
225                result.status,
226                stderr.chars().take(500).collect::<String>()
227            )));
228        }
229
230        // Read extracted WAV
231        let wav_data = tokio::fs::read(&output_path).await.map_err(|e| {
232            VideoParserError::IoError(format!("Failed to read extracted audio: {}", e))
233        })?;
234
235        let _ = tokio::fs::remove_file(&output_path).await;
236
237        if wav_data.is_empty() {
238            return Err(VideoParserError::FfmpegError(
239                "ffmpeg produced empty audio output (video may have no audio track)".to_string(),
240            ));
241        }
242
243        Ok(wav_data)
244    }

246    /// Parse video from file path
247    pub async fn parse_file(&self, path: &Path) -> Result<VideoParseResult, VideoParserError> {
248        let format = path
249            .extension()
250            .and_then(|e| e.to_str())
251            .ok_or(VideoParserError::InvalidPath)?;
252
253        let video_data =
254            std::fs::read(path).map_err(|e| VideoParserError::IoError(e.to_string()))?;
255
256        self.parse(&video_data, format).await
257    }

266#[async_trait]
267impl DocumentParser for VideoParser {
268    fn supported_extensions(&self) -> &[&str] {
269        &["mp4", "mov", "avi", "mkv", "webm", "wmv", "flv", "m4v"]
270    }
271
272    async fn parse(&self, data: &[u8], filename: &str) -> crate::Result<Vec<TextChunk>> {
273        let ext = std::path::Path::new(filename)
274            .extension()
275            .and_then(|e| e.to_str())
276            .unwrap_or("mp4");
277
278        let result = self.parse(data, ext).await.map_err(|e| {
279            cogkos_core::CogKosError::Parse(format!("Video parsing failed for {}: {}", filename, e))
280        })?;
281
282        let mut chunks = Vec::new();
283
284        if let Some(ref transcription) = result.audio_transcription
285            && !transcription.text.is_empty()
286        {
287            let mut metadata = std::collections::HashMap::new();
288            metadata.insert("source".to_string(), filename.to_string());
289            metadata.insert("type".to_string(), "video_transcription".to_string());
290            metadata.insert(
291                "duration_secs".to_string(),
292                format!("{:.1}", result.duration_secs),
293            );
294            metadata.insert("width".to_string(), result.width.to_string());
295            metadata.insert("height".to_string(), result.height.to_string());
296
297            chunks.push(TextChunk {
298                content: transcription.text.clone(),
299                chunk_index: 0,
300                metadata,
301            });
302        }
303
304        if chunks.is_empty() {
305            tracing::warn!("No text content extracted from video: {}", filename);
306        }
307
308        Ok(chunks)
309    }
