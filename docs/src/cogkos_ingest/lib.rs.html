32/// Document type detected from content
33#[derive(Debug, Clone)]
34pub enum DocumentType {
35    Pdf,
36    Word,
37    Markdown,
38    Text,
39    Excel,
40    PowerPoint,
41    Csv,
42    Unknown(String),
43}
44
45impl DocumentType {
46    /// Detect from filename extension
47    pub fn from_filename(filename: &str) -> Self {
48        let lower = filename.to_lowercase();
49        if lower.ends_with(".pdf") {
50            DocumentType::Pdf
51        } else if lower.ends_with(".docx") || lower.ends_with(".doc") {
52            DocumentType::Word
53        } else if lower.ends_with(".md") || lower.ends_with(".markdown") {
54            DocumentType::Markdown
55        } else if lower.ends_with(".txt") {
56            DocumentType::Text
57        } else if lower.ends_with(".xlsx") || lower.ends_with(".xls") || lower.ends_with(".csv") {
58            DocumentType::Excel
59        } else if lower.ends_with(".pptx") || lower.ends_with(".ppt") {
60            DocumentType::PowerPoint
61        } else {
62            DocumentType::Unknown(lower.split('.').next_back().unwrap_or("").to_string())
63        }
64    }
65
66    /// Get MIME type
67    pub fn mime_type(&self) -> &'static str {
68        match self {
69            DocumentType::Pdf => "application/pdf",
70            DocumentType::Word => {
71                "application/vnd.openxmlformats-officedocument.wordprocessingml.document"
72            }
73            DocumentType::Markdown => "text/markdown",
74            DocumentType::Text => "text/plain",
75            DocumentType::Excel => {
76                "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
77            }
78            DocumentType::PowerPoint => {
79                "application/vnd.openxmlformats-officedocument.presentationml.presentation"
80            }
81            DocumentType::Csv => "text/csv",
82            DocumentType::Unknown(_) => "application/octet-stream",
83        }
84    }
85}
