1//! LLM-powered deep classification for documents
2//!
3//! This module provides LLM-based deep classification that extracts:
4//! - Industry information
5//! - Company/entity information
6//! - Document type
7//! - Key conclusions
8//! - Predictions
9//! - Data points
10//! - Methodological insights

14/// LLM Client trait for deep classification
15/// This is a simplified trait - actual implementation would use cogkos-llm
16pub trait DeepClassificationLlmClient: Send + Sync {
17    /// Generate a completion from the LLM
18    fn generate(&self, prompt: &str) -> DeepClassificationResult;
19}

21/// Result type for LLM generation
22pub struct DeepClassificationResult {
23    pub content: String,
24    pub confidence: f64,
25}

27/// Deep classification result
28#[derive(Debug, Clone, Serialize, Deserialize)]
29pub struct DeepClassification {
30    /// Industry sector (e.g., "Manufacturing", "Retail", "Finance")
31    pub industry: Option<String>,
32    /// Companies or entities mentioned
33    pub entities: Vec<EntityMention>,
34    /// Document type classification
35    pub document_type: Option<String>,
36    /// Key conclusions extracted
37    pub conclusions: Vec<String>,
38    /// Predictions mentioned in the document
39    pub predictions: Vec<Prediction>,
40    /// Key data points
41    pub data_points: Vec<DataPoint>,
42    /// Methodological insights
43    pub methodologies: Vec<String>,
44    /// Overall confidence in the classification
45    pub confidence: f64,
46}

114    /// Extract deep classification from document content
115    ///
116    /// # Arguments
117    /// * `content` - Full document content
118    /// * `llm_client` - Optional LLM client for extraction
119    ///
120    /// # Returns
121    /// DeepClassification with extracted information
122    pub async fn classify(
123        &self,
124        content: &str,
125        llm_client: Option<&dyn DeepClassificationLlmClient>,
126    ) -> DeepClassification {
127        if self.config.enable_llm
128            && let Some(client) = llm_client
129        {
130            return self.classify_with_llm(content, client).await;
131        }
132
133        // Fallback to rule-based extraction
134        self.classify_rule_based(content)
135    }

137    /// LLM-based deep classification
138    async fn classify_with_llm(
139        &self,
140        content: &str,
141        client: &dyn DeepClassificationLlmClient,
142    ) -> DeepClassification {
143        // Build prompt for extraction
144        let prompt = format!(
145            r#"Extract structured information from the following document.
146Respond in JSON format with these fields:
147- industry: Main industry sector (or null if unclear)
148- entities: Array of {{name, type}} for companies, products, persons
149- document_type: Type of document (report, analysis, paper, etc.)
150- conclusions: Array of key conclusions
151- predictions: Array of {{content, confidence, time_horizon}} for predictions
152- data_points: Array of {{label, value, unit, context}} for key metrics
153- methodologies: Array of methodological approaches mentioned
154
155Document content:
156{}
157
158Respond only with valid JSON."#,
159            content.chars().take(5000).collect::<String>()
160        );
161
162        // Call LLM
163        let result = client.generate(&prompt);
164
165        // Parse JSON response
166        match serde_json::from_str::<serde_json::Value>(&result.content) {
167            Ok(json) => {
168                let industry = json
169                    .get("industry")
170                    .and_then(|v| v.as_str())
171                    .map(String::from);
172                let document_type = json
173                    .get("document_type")
174                    .and_then(|v| v.as_str())
175                    .map(String::from);
176
177                let entities: Vec<EntityMention> = json
178                    .get("entities")
179                    .and_then(|v| v.as_array())
180                    .map(|arr| {
181                        arr.iter()
182                            .filter_map(|e| {
183                                let name = e.get("name")?.as_str()?.to_string();
184                                let entity_type = match e.get("type").and_then(|v| v.as_str()) {
185                                    Some("Company") => EntityType::Company,
186                                    Some("Person") => EntityType::Person,
187                                    Some("Product") => EntityType::Product,
188                                    Some("Location") => EntityType::Location,
189                                    Some("Organization") => EntityType::Organization,
190                                    _ => EntityType::Other,
191                                };
192                                Some(EntityMention { name, entity_type })
193                            })
194                            .collect()
195                    })
196                    .unwrap_or_default();
197
198                let conclusions: Vec<String> = json
199                    .get("conclusions")
200                    .and_then(|v| v.as_array())
201                    .map(|arr| {
202                        arr.iter()
203                            .filter_map(|v| v.as_str())
204                            .map(String::from)
205                            .collect()
206                    })
207                    .unwrap_or_default();
208
209                let predictions: Vec<Prediction> = json
210                    .get("predictions")
211                    .and_then(|v| v.as_array())
212                    .map(|arr| {
213                        arr.iter()
214                            .filter_map(|p| {
215                                let content = p.get("content")?.as_str()?.to_string();
216                                let confidence =
217                                    p.get("confidence").and_then(|v| v.as_f64()).unwrap_or(0.0);
218                                let time_horizon = p
219                                    .get("time_horizon")
220                                    .and_then(|v| v.as_str())
221                                    .map(String::from);
222                                Some(Prediction {
223                                    content,
224                                    confidence,
225                                    time_horizon,
226                                })
227                            })
228                            .collect()
229                    })
230                    .unwrap_or_default();
231
232                let data_points: Vec<DataPoint> = json
233                    .get("data_points")
234                    .and_then(|v| v.as_array())
235                    .map(|arr| {
236                        arr.iter()
237                            .filter_map(|d| {
238                                let label = d.get("label")?.as_str()?.to_string();
239                                let value = d.get("value")?.as_str()?.to_string();
240                                let unit = d.get("unit").and_then(|v| v.as_str()).map(String::from);
241                                let context = d.get("context")?.as_str()?.to_string();
242                                Some(DataPoint {
243                                    label,
244                                    value,
245                                    unit,
246                                    context,
247                                })
248                            })
249                            .collect()
250                    })
251                    .unwrap_or_default();
252
253                let methodologies: Vec<String> = json
254                    .get("methodologies")
255                    .and_then(|v| v.as_array())
256                    .map(|arr| {
257                        arr.iter()
258                            .filter_map(|v| v.as_str())
259                            .map(String::from)
260                            .collect()
261                    })
262                    .unwrap_or_default();
263
264                DeepClassification {
265                    industry,
266                    entities,
267                    document_type,
268                    conclusions,
269                    predictions,
270                    data_points,
271                    methodologies,
272                    confidence: result.confidence,
273                }
274            }
275            Err(_) => {
276                // Fallback to rule-based if JSON parsing fails
277                self.classify_rule_based(content)
278            }
279        }
280    }

282    /// Rule-based deep classification (fallback)
283    fn classify_rule_based(&self, content: &str) -> DeepClassification {
284        let content_lower = content.to_lowercase();
285
286        // Extract industry
287        let industry = self.extract_industry(&content_lower);
288
289        // Extract entities
290        let entities = self.extract_entities(content);
291
292        // Extract document type
293        let document_type = self.extract_document_type(&content_lower);
294
295        // Extract conclusions
296        let conclusions = self.extract_conclusions(content);
297
298        // Extract predictions
299        let predictions = self.extract_predictions(&content_lower);
300
301        // Extract data points
302        let data_points = self.extract_data_points(content);
303
304        // Extract methodologies
305        let methodologies = self.extract_methodologies(&content_lower);
306
307        DeepClassification {
308            industry,
309            entities,
310            document_type,
311            conclusions,
312            predictions,
313            data_points,
314            methodologies,
315            confidence: 0.6, // Rule-based has lower confidence
316        }
317    }

319    fn extract_industry(&self, content: &str) -> Option<String> {
320        // 中文关键词优先匹配
321        let chinese_industries = [
322            ("制造业", "制造业"),
323            ("生产制造", "制造业"),
324            ("零售", "零售业"),
325            ("金融", "金融业"),
326            ("银行", "金融业"),
327            ("医疗", "医疗健康"),
328            ("健康", "医疗健康"),
329            ("科技", "科技业"),
330            ("软件", "软件业"),
331            ("能源", "能源业"),
332            ("汽车制造", "汽车制造业"),
333        ];
334
335        for (keyword, industry) in chinese_industries {
336            if content.contains(keyword) {
337                return Some(industry.to_string());
338            }
339        }
340
341        // 英文关键词匹配
342        let industries = [
343            ("manufacturing", "制造业"),
344            ("manufacture", "制造业"),
345            ("retail", "零售业"),
346            ("finance", "金融业"),
347            ("banking", "金融业"),
348            ("healthcare", "医疗健康"),
349            ("technology", "科技业"),
350            ("software", "软件业"),
351            ("energy", "能源业"),
352            ("automotive", "汽车制造业"),
353        ];
354
355        for (keyword, industry) in industries {
356            if content.contains(keyword) {
357                return Some(industry.to_string());
358            }
359        }
360        None
361    }

363    fn extract_entities(&self, content: &str) -> Vec<EntityMention> {
364        let mut entities = Vec::new();
365
366        // Simple entity extraction patterns
367        // Company patterns
368        let company_patterns = ["公司", "集团", "企业", "Inc", "Corp", "Ltd"];
369        for pattern in company_patterns {
370            if content.contains(pattern) {
371                // Simplified extraction
372                entities.push(EntityMention {
373                    name: pattern.to_string(),
374                    entity_type: EntityType::Company,
375                });
376            }
377        }
378
379        entities.truncate(self.config.max_entities);
380        entities
381    }

383    fn extract_document_type(&self, content: &str) -> Option<String> {
384        if content.contains("年报") || content.contains("annual report") {
385            Some("年度报告".to_string())
386        } else if content.contains("季报") || content.contains("quarterly") {
387            Some("季度报告".to_string())
388        } else if content.contains("分析") || content.contains("analysis") {
389            Some("分析报告".to_string())
390        } else if content.contains("研究") || content.contains("research") {
391            Some("研究报告".to_string())
392        } else {
393            Some("文档".to_string())
394        }
395    }

397    fn extract_conclusions(&self, content: &str) -> Vec<String> {
398        let mut conclusions = Vec::new();
399
400        // Look for conclusion markers
401        let markers = ["结论", "结论如下", "总结", "综上", "因此", "总之"];
402

413    fn extract_predictions(&self, content: &str) -> Vec<Prediction> {
414        let mut predictions = Vec::new();
415
416        // Look for prediction markers
417        let markers = ["预计", "预测", "预期", "将", "will", "expect", "forecast"];
418
419        for marker in markers {
420            if content.contains(marker) {
421                predictions.push(Prediction {
422                    content: marker.to_string(),
423                    confidence: 0.5,
424                    time_horizon: None,
425                });
426            }
427        }
428
429        predictions
430    }

432    fn extract_data_points(&self, content: &str) -> Vec<DataPoint> {
433        // Look for numeric patterns that might be data points
434        let mut data_points = Vec::new();
435
436        // Simple percentage extraction
437        let percent_matches: Vec<_> = content.match_indices("%").collect();
438        for (pos, _) in percent_matches.iter().take(10) {
439            let start = pos.saturating_sub(20);
440            let end = (pos + 1).min(content.len());
441            let context = &content[start..end];
442
443            data_points.push(DataPoint {
444                label: "percentage".to_string(),
445                value: context.to_string(),
446                unit: Some("%".to_string()),
447                context: context.to_string(),
448            });
449        }
450
451        data_points
452    }

483/// Convert deep classification to graph relations
484impl DeepClassification {
485    /// Generate graph relation suggestions based on classification
486    pub fn to_graph_relations(&self, file_claim_id: &str) -> Vec<GraphRelationSuggestion> {
487        let mut relations = Vec::new();
488
489        // File -> Industry
490        if let Some(ref industry) = self.industry {
491            relations.push(GraphRelationSuggestion {
492                from_id: file_claim_id.to_string(),
493                to_label: industry.clone(),
494                relation_type: "INDUSTRY".to_string(),
495            });
496        }
497
498        // File -> Entity
499        for entity in &self.entities {
500            relations.push(GraphRelationSuggestion {
501                from_id: file_claim_id.to_string(),
502                to_label: entity.name.clone(),
503                relation_type: "ABOUT".to_string(),
504            });
505        }
506
507        // File -> Conclusions
508        for conclusion in &self.conclusions {
509            relations.push(GraphRelationSuggestion {
510                from_id: file_claim_id.to_string(),
511                to_label: conclusion.clone(),
512                relation_type: "CONTAINS".to_string(),
513            });
514        }
515
516        // File -> Predictions
517        for prediction in &self.predictions {
518            relations.push(GraphRelationSuggestion {
519                from_id: file_claim_id.to_string(),
520                to_label: prediction.content.clone(),
521                relation_type: "CONTAINS_PREDICTION".to_string(),
522            });
523        }
524
525        relations
526    }
527}
