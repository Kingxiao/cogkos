15/// Submit experience handler
16pub async fn handle_submit_experience(
17    req: SubmitExperienceRequest,
18    tenant_id: &str,
19    claim_store: Arc<dyn ClaimStore>,
20    vector_store: Arc<dyn VectorStore>,
21    graph_store: Arc<dyn GraphStore>,
22    embedding_client: Option<Arc<dyn LlmClient>>,
23) -> Result<serde_json::Value> {
24    // Convert source
25    let claimant = match req.source {
26        SourceInfo::Human { user_id } => Claimant::Human {
27            user_id,
28            role: "user".to_string(),
29        },
30        SourceInfo::Agent { agent_id, model } => Claimant::Agent { agent_id, model },
31        SourceInfo::External { source_name } => Claimant::ExternalPublic { source_name },
32    };
33
34    // Extract domain before consuming req.content
35    let domain = extract_domain(&req.content);
36
37    let provenance = ProvenanceRecord {
38        source_id: match &claimant {
39            Claimant::Agent { agent_id, .. } => agent_id.clone(),
40            Claimant::Human { user_id, .. } => user_id.clone(),
41            Claimant::ExternalPublic { source_name } => source_name.clone(),
42            Claimant::System => "system".to_string(),
43        },
44        source_type: "experience".to_string(),
45        ingestion_method: "mcp_submit".to_string(),
46        original_url: None,
47        audit_hash: calculate_hash(&req.content),
48    };
49
50    let access_envelope = AccessEnvelope::new(tenant_id).with_visibility(Visibility::Tenant);
51
52    let mut claim = EpistemicClaim::new(
53        req.content,
54        tenant_id,
55        req.node_type,
56        claimant,
57        access_envelope,
58        provenance,
59    );
60
61    claim.confidence = req.confidence.unwrap_or(0.5);
62    claim.t_valid_start = req.valid_from.unwrap_or(claim.t_valid_start);
63    claim.t_valid_end = req.valid_to;
64    claim.derived_from = req.related_to;
65
66    // Save domain to metadata
67    claim
68        .metadata
69        .insert("domain".to_string(), serde_json::Value::String(domain));
70
71    // Memory layer + session scoping
72    if let Some(ref layer) = req.memory_layer {
73        claim.metadata.insert(
74            "memory_layer".to_string(),
75            serde_json::Value::String(layer.clone()),
76        );
77    }
78    if let Some(ref sid) = req.session_id {
79        claim.metadata.insert(
80            "session_id".to_string(),
81            serde_json::Value::String(sid.clone()),
82        );
83    }
84
85    let t0 = std::time::Instant::now();
86    let claim_id = claim_store.insert_claim(&claim).await?;
87    tracing::info!(claim_id = %claim_id, pg_ms = t0.elapsed().as_millis() as u64, "Fast-track: claim persisted");
88
89    // S2: Fast capture / slow consolidation
90    // Return immediately after PG write, spawn background task for embedding + vector + graph + conflict
91    let bg_claim = claim.clone();
92    let bg_tenant = tenant_id.to_string();
93    let bg_claim_id = claim_id;
94    let bg_claim_store = Arc::clone(&claim_store);
95    let bg_vector_store = Arc::clone(&vector_store);
96    let bg_graph_store = Arc::clone(&graph_store);
97
98    tokio::spawn(async move {
99        let claim_store = bg_claim_store;
100        let vector_store = bg_vector_store;
101        let graph_store = bg_graph_store;
102        let t_bg = std::time::Instant::now();
103
104        // 1. Embedding
105        let content_vector: Vec<f32> = if let Some(ref client) = embedding_client {
106            let embedding_service = EmbeddingService::new(client.clone());
107            match embedding_service.embed(&bg_claim.content).await {
108                Ok(vec) => vec,
109                Err(e) => {
110                    tracing::warn!(claim_id = %bg_claim_id, error = %e, "Background embedding failed, using fallback");
111                    generate_query_vector(&bg_claim.content, helpers::DEFAULT_FALLBACK_DIM)
112                }
113            }
114        } else {
115            generate_query_vector(&bg_claim.content, helpers::DEFAULT_FALLBACK_DIM)
116        };
117        let t_embed = t_bg.elapsed();
118
119        // 2. Vector upsert
120        let payload = serde_json::json!({
121            "tenant_id": bg_tenant,
122            "content": bg_claim.content,
123            "node_type": bg_claim.node_type.as_db_str(),
124        });
125        if let Err(e) = vector_store
126            .upsert(bg_claim_id, content_vector.clone(), payload)
127            .await
128        {
129            tracing::warn!(claim_id = %bg_claim_id, error = %e, "Background vector upsert failed");
130            // Mark for sleep-time reindex
131            let mut patched = bg_claim.clone();
132            patched
133                .metadata
134                .insert("needs_reindex".into(), serde_json::Value::Bool(true));
135            claim_store.update_claim(&patched).await.ok();
136        }
137
138        // 3. Graph upsert
139        if let Err(e) = graph_store.upsert_node(&bg_claim).await {
140            tracing::warn!(claim_id = %bg_claim_id, error = %e, "Background graph upsert failed");
141        }
142
143        // 4. Semantic conflict detection + graph edge building
144        if let Ok(matches) = vector_store.search(content_vector, &bg_tenant, 20).await {
145            let scored_matches: Vec<_> = matches
146                .into_iter()
147                .filter(|m| m.id != bg_claim_id && m.score > 0.3)
148                .take(10)
149                .collect();
150
151            if !scored_matches.is_empty() {
152                let mut candidate_claims: Vec<EpistemicClaim> = Vec::new();
153                for m in &scored_matches {
154                    if let Ok(c) = claim_store.get_claim(m.id, &bg_tenant).await {
155                        candidate_claims.push(c);
156                    }
157                }
158
159                for m in &scored_matches {
160                    if m.score > 0.5 {
161                        let relation = if m.score > 0.8 { "SIMILAR" } else { "RELATED" };
162                        graph_store
163                            .create_edge(bg_claim_id, m.id, relation, m.score)
164                            .await
165                            .ok();
166                    }
167                }
168
169                let conflicts = cogkos_core::evolution::conflict::detect_conflicts_batch(
170                    &bg_claim,
171                    &candidate_claims,
172                );
173                for conflict in &conflicts {
174                    claim_store.insert_conflict(conflict).await.ok();
175                }
176
177                tracing::debug!(
178                    claim_id = %bg_claim_id,
179                    edges = scored_matches.iter().filter(|m| m.score > 0.5).count(),
180                    conflicts = conflicts.len(),
181                    "Background: graph edges + conflict detection"
182                );
183            }
184        }
185
186        tracing::info!(
187            claim_id = %bg_claim_id,
188            embedding_ms = t_embed.as_millis() as u64,
189            total_bg_ms = t_bg.elapsed().as_millis() as u64,
190            "Background consolidation complete"
191        );
192    });
193
194    let resp = serde_json::json!({
195        "claim_id": claim_id.to_string(),
196        "status": "accepted",
197        "consolidation": "async",
198        "estimated_consolidation_time": "5s"
199    });
200    Ok(resp)
201}

203/// Upload document handler
204pub async fn handle_upload_document(
205    req: UploadDocumentRequest,
206    tenant_id: &str,
207    claim_store: &dyn ClaimStore,
208    graph_store: &dyn GraphStore,
209    vector_store: &dyn VectorStore,
210    object_store: &dyn ObjectStore,
211    embedding_service: Option<EmbeddingService>,
212) -> Result<DocumentUploadResponse> {
213    // Decode base64 content
214    let file_data =
215        match base64::Engine::decode(&base64::engine::general_purpose::STANDARD, &req.content) {
216            Ok(data) => data,
217            Err(e) => return Err(CogKosError::InvalidInput(format!("Invalid base64: {}", e))),
218        };
219
220    // Calculate content hash for duplicate detection
221    let content_hash = calculate_content_hash(&file_data);
222
223    // Check for duplicate using hash-based key
224    let hash_key = format!("{}/by_hash/{}", tenant_id, content_hash);
225
226    if object_store.download(&hash_key).await.is_ok() {
227        tracing::info!("Duplicate document detected, hash: {}", &content_hash[..16]);
228        return Ok(DocumentUploadResponse {
229            file_id: content_hash[..8].to_string(),
230            status: "duplicate".to_string(),
231            estimated_time: "0ms".to_string(),
232            pipeline_id: None,
233            is_duplicate: true,
234        });
235    }
236
237    // Determine content type from filename
238    let content_type = match req.filename.rsplit('.').next() {
239        Some("pdf") => "application/pdf",
240        Some("docx") => "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
241        Some("md") => "text/markdown",
242        Some("txt") => "text/plain",
243        Some("csv") => "text/csv",
244        Some("xlsx") => "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
245        Some("pptx") => "application/vnd.openxmlformats-officedocument.presentationml.presentation",
246        _ => "application/octet-stream",
247    };
248
249    let file_id = uuid::Uuid::new_v4();
250
251    let claimant = match req.source {
252        SourceInfo::Human { user_id } => Claimant::Human {
253            user_id,
254            role: "user".to_string(),
255        },
256        SourceInfo::Agent { agent_id, model } => Claimant::Agent { agent_id, model },
257        SourceInfo::External { source_name } => Claimant::ExternalPublic { source_name },
258    };
259
260    let hash_upload_key = format!("{}/by_hash/{}", tenant_id, content_hash);
261
262    let (status, pipeline_id, estimated_time) = if req.auto_process {
263        if let Some(embedding_svc) = embedding_service {
264            if let Err(e) = object_store
265                .upload(&hash_upload_key, &file_data, content_type)
266                .await
267            {
268                tracing::warn!(key = %hash_upload_key, error = %e, "Failed to store hash-keyed duplicate detection copy");
269            }
270
271            let uploaded_file = UploadedFile {
272                filename: req.filename.clone(),
273                content_type: content_type.to_string(),
274                data: file_data,
275                source: claimant,
276                tenant_id: tenant_id.to_string(),
277            };
278
279            let pipeline = IngestionPipeline::new(embedding_svc.clone());
280
281            match pipeline
282                .ingest(
283                    uploaded_file,
284                    claim_store,
285                    graph_store,
286                    vector_store,
287                    object_store,
288                )
289                .await
290            {
291                Ok(result) => (
292                    "completed",
293                    Some(format!("pipe-{}", &file_id.to_string()[..8])),
294                    format!("{}ms", result.chunk_claim_ids.len() * 100),
295                ),
296                Err(e) => {
297                    tracing::error!("Ingestion pipeline error: {}", e);
298                    (
299                        "failed",
300                        Some(format!("pipe-{}", &file_id.to_string()[..8])),
301                        "0s".to_string(),
302                    )
303                }
304            }
305        } else {
306            let s3_key = format!("{}/raw/{}/{}", tenant_id, file_id, req.filename);
307            object_store
308                .upload(&s3_key, &file_data, content_type)
309                .await?;
310            if let Err(e) = object_store
311                .upload(&hash_upload_key, &file_data, content_type)
312                .await
313            {
314                tracing::warn!(key = %hash_upload_key, error = %e, "Failed to store hash-keyed duplicate detection copy");
315            }
316
317            ("uploaded_no_processing", None, "0s".to_string())
318        }
319    } else {
320        let s3_key = format!("{}/raw/{}/{}", tenant_id, file_id, req.filename);
321        object_store
322            .upload(&s3_key, &file_data, content_type)
323            .await?;
324        if let Err(e) = object_store
325            .upload(&hash_upload_key, &file_data, content_type)
326            .await
327        {
328            tracing::warn!(key = %hash_upload_key, error = %e, "Failed to store hash-keyed duplicate detection copy");
329        }
330
331        let domain = extract_domain(&req.filename);
332
333        let provenance = ProvenanceRecord {
334            source_id: "upload".to_string(),
335            source_type: "document".to_string(),
336            ingestion_method: "mcp_upload".to_string(),
337            original_url: Some(format!("s3://{}", s3_key)),
338            audit_hash: content_hash.clone(),
339        };
340
341        let access_envelope = AccessEnvelope::new(tenant_id).with_visibility(Visibility::Tenant);
342
343        let mut claim = EpistemicClaim::new(
344            format!("Document: {}", req.filename),
345            tenant_id,
346            NodeType::File,
347            Claimant::System,
348            access_envelope,
349            provenance,
350        );
351
352        claim
353            .metadata
354            .insert("domain".to_string(), serde_json::Value::String(domain));
355        claim.metadata.insert(
356            "content_hash".to_string(),
357            serde_json::Value::String(content_hash),
358        );
359
360        let _claim_id = claim_store.insert_claim(&claim).await?;
361
362        ("uploaded", None, "0s".to_string())
363    };
364
365    Ok(DocumentUploadResponse {
366        file_id: file_id.to_string(),
367        status: status.to_string(),
368        estimated_time,
369        pipeline_id,
370        is_duplicate: false,
371    })
372}

374/// Get meta directory handler
375#[allow(clippy::type_complexity)]
376pub async fn handle_get_meta_directory(
377    req: GetMetaDirectoryRequest,
378    tenant_id: &str,
379    claim_store: &dyn ClaimStore,
380) -> Result<serde_json::Value> {
381    let claims = claim_store
382        .query_claims(tenant_id, &[])
383        .await
384        .map_err(|e| CogKosError::Database(format!("Failed to query claims: {}", e)))?;
385
386    if claims.is_empty() {
387        return Ok(serde_json::json!({
388            "entries": [],
389            "total_domains": 0,
390            "total_claims": 0
391        }));
392    }
393
394    let mut domain_map: std::collections::HashMap<
395        String,
396        (
397            Vec<cogkos_core::models::EpistemicClaim>,
398            std::collections::HashMap<String, usize>,
399        ),
400    > = std::collections::HashMap::new();
401
402    for claim in &claims {
403        let domain = claim
404            .metadata
405            .get("domain")
406            .and_then(|v| v.as_str())
407            .unwrap_or("general")
408            .to_string();
409
410        let entry = domain_map
411            .entry(domain)
412            .or_insert((Vec::new(), std::collections::HashMap::new()));
413        entry.0.push(claim.clone());
414
415        let node_type_str = format!("{:?}", claim.node_type);
416        *entry.1.entry(node_type_str).or_insert(0) += 1;
417    }
418
419    let filtered_domains: Vec<(
420        String,
421        (
422            Vec<cogkos_core::models::EpistemicClaim>,
423            std::collections::HashMap<String, usize>,
424        ),
425    )> = if let Some(ref query_domain) = req.query_domain {
426        domain_map
427            .into_iter()
428            .filter(|(d, _)| d == query_domain)
429            .collect()
430    } else {
431        domain_map.into_iter().collect()
432    };
433
434    let mut entries: Vec<MetaDirectoryEntry> = filtered_domains
435        .into_iter()
436        .map(|(domain, (claims_in_domain, node_types))| {
437            let claim_count = claims_in_domain.len();
438            let avg_confidence: f64 = if claim_count > 0 {
439                claims_in_domain.iter().map(|c| c.confidence).sum::<f64>() / claim_count as f64
440            } else {
441                0.0
442            };
443
444            let expertise_score = (claim_count as f64 * avg_confidence / 10.0).min(1.0);
445            let latest_update = claims_in_domain.iter().map(|c| c.updated_at).max();
446
447            MetaDirectoryEntry {
448                domain,
449                claim_count,
450                expertise_score,
451                node_types,
452                avg_confidence,
453                latest_update,
454            }
455        })
456        .collect();
457
458    entries.sort_by(|a, b| {
459        b.expertise_score
460            .partial_cmp(&a.expertise_score)
461            .unwrap_or(std::cmp::Ordering::Equal)
462    });
463
464    if let Some(min_score) = req.min_expertise_score {
465        entries.retain(|e| e.expertise_score >= min_score);
466    }
467
468    let total_claims: usize = entries.iter().map(|e| e.claim_count).sum();
469    let total_domains = entries.len();
470
471    Ok(serde_json::json!({
472        "entries": entries,
473        "total_domains": total_domains,
474        "total_claims": total_claims
475    }))
