15/// Query knowledge handler
16#[allow(clippy::too_many_arguments)]
17pub async fn handle_query_knowledge(
18    req: QueryKnowledgeRequest,
19    tenant_id: &str,
20    _roles: &[String],
21    claim_store: &dyn ClaimStore,
22    vector_store: &dyn VectorStore,
23    graph_store: &dyn GraphStore,
24    cache_store: &dyn CacheStore,
25    gap_store: &dyn GapStore,
26    llm_client: Option<Arc<dyn LlmClient>>,
27    embedding_client: Option<Arc<dyn LlmClient>>,
28) -> Result<McpQueryResponse> {
29    let start_time = std::time::Instant::now();
30
31    // Validate query is not empty
32    if req.query.trim().is_empty() {
33        return Ok(McpQueryResponse {
34            query_hash: 0,
35            query_context: req.query.clone(),
36            best_belief: None,
37            related_by_graph: vec![],
38            conflicts: vec![],
39            prediction: None,
40            knowledge_gaps: vec![],
41            freshness: FreshnessInfo {
42                newest_source: None,
43                oldest_source: None,
44                staleness_warning: false,
45            },
46            cache_status: CacheStatus::Miss,
47            cognitive_path: None,
48            metadata: QueryMetadata {
49                execution_time_ms: 0,
50                cache_hit_rate: 0.0,
51                processed_claims: 0,
52                related_node_count: 0,
53                conflict_count: 0,
54            },
55        });
56    }
57
58    // Calculate query hash for cache lookup
59    let query_hash = calculate_query_hash(&req.query, &req.context.domain);
60
61    // 1. Check cache — S6 dual-path decision (skip if high urgency)
62    let dual_path = DualPathThresholds::from_env();
63    if !matches!(req.context.urgency, Urgency::High)
64        && let Some(cached) = cache_store.get_cached(tenant_id, query_hash).await?
65    {
66        let ttl_seconds = std::env::var("CACHE_TTL_SECONDS")
67            .ok()
68            .and_then(|s| s.parse::<i64>().ok())
69            .unwrap_or(3600);
70
71        if cached.is_valid(ttl_seconds) {
72            // S6: Determine cognitive path
73            if cached.qualifies_for_system1(&dual_path) {
74                // === System 1: Fast path — high confidence, return immediately ===
75                cache_store.record_hit(tenant_id, query_hash).await?;
76
77                // S3 读即写: update activation for cached claims
78                if let Some(ref belief) = cached.response.best_belief {
79                    for claim_id in &belief.claim_ids {
80                        claim_store
81                            .update_activation(*claim_id, tenant_id, 0.05)
82                            .await
83                            .ok();
84                    }
85                }
86
87                let hit_rate = cached.success_rate();
88                let mut response = cached.response.clone();
89                response.cache_status = CacheStatus::Hit;
90                response.cognitive_path = Some(CognitivePath::System1);
91                response.metadata.execution_time_ms = start_time.elapsed().as_millis() as u64;
92                response.metadata.cache_hit_rate = hit_rate;
93
94                tracing::debug!(
95                    query_hash = query_hash,
96                    confidence = cached.confidence,
97                    hit_count = cached.hit_count,
98                    success_rate = hit_rate,
99                    "System 1 fast path: cache hit with high confidence"
100                );
101
102                return Ok(response);
103            } else {
104                // Cache exists but low confidence → degrade to System 2
105                tracing::info!(
106                    query_hash = query_hash,
107                    confidence = cached.confidence,
108                    success_rate = cached.success_rate(),
109                    "System 2 triggered: cache confidence below threshold, running full reasoning"
110                );
111                // Fall through to System 2 full path
112            }
113        } else {
114            tracing::debug!(
115                query_hash = query_hash,
116                confidence = cached.confidence,
117                invalidated = cached.invalidated_by.is_some(),
118                "Cache entry expired or invalid, using System 2"
119            );
120        }
121    }
122
123    // 2. Vector search - embed query using embedding service
124    let query_vector = if let Some(ref client) = embedding_client {
125        let embedding_service = EmbeddingService::new(client.clone());
126        match embedding_service.embed(&req.query).await {
127            Ok(vec) => vec,
128            Err(e) => {
129                tracing::warn!("Embedding failed, using fallback: {}", e);
130                generate_query_vector(&req.query, helpers::DEFAULT_FALLBACK_DIM)
131            }
132        }
133    } else {
134        tracing::warn!("No embedding client configured, using fallback");
135        generate_query_vector(&req.query, helpers::DEFAULT_FALLBACK_DIM)
136    };
137    let vector_matches = vector_store
138        .search(query_vector, tenant_id, req.context.max_results)
139        .await?;
140
141    // 3. Get claims from vector matches with comprehensive permission filtering
142    let mut claims = Vec::new();
143    let mut claim_ids: Vec<uuid::Uuid> = Vec::new();
144    for m in &vector_matches {
145        if let Ok(claim) = claim_store.get_claim(m.id, tenant_id).await {
146            claim_ids.push(claim.id);
147            claims.push(claim);
148        }
149    }
150
151    // 3b. Filter by memory_layer / session_id if requested
152    if let Some(ref layer) = req.memory_layer {
153        claims.retain(|c| {
154            c.metadata
155                .get("memory_layer")
156                .and_then(|v| v.as_str())
157                .is_some_and(|l| l == layer)
158        });
159    }
160    if let Some(ref sid) = req.session_id {
161        claims.retain(|c| {
162            c.metadata
163                .get("session_id")
164                .and_then(|v| v.as_str())
165                .is_some_and(|s| s == sid)
166        });
167    }
168    if req.memory_layer.is_some() || req.session_id.is_some() {
169        claim_ids = claims.iter().map(|c| c.id).collect();
170    }
171
172    // 3c. Record rehearsal for retrieved claims (S3: read equals write)
173    for claim in &claims {
174        let layer = cogkos_core::models::MemoryLayer::from_metadata(&claim.metadata);
175        let delta = layer.lambda() * 0.4;
176        claim_store
177            .update_activation(claim.id, tenant_id, delta)
178            .await
179            .ok();
180    }
181
182    // 4. Graph activation diffusion with threshold
183    let mut all_graph_nodes: Vec<GraphNode> = Vec::new();
184    let threshold = req.activation_threshold.clamp(0.0, 1.0);
185    let _decay_factor = 0.8;
186    let _max_depth = 2;
187
188    for claim in &claims {
189        let related = match graph_store.find_related(claim.id, 2, threshold).await {
190            Ok(nodes) => nodes,
191            Err(e) => {
192                tracing::warn!(claim_id = %claim.id, error = %e, "Graph diffusion failed, degrading");
193                vec![]
194            }
195        };
196        for mut node in related {
197            if !all_graph_nodes
198                .iter()
199                .any(|n: &GraphNode| n.content == node.content)
200            {
201                // Enrich with live activation from PG (FalkorDB stores stale initial value)
202                if let Ok(live_claim) = claim_store.get_claim(node.id, tenant_id).await {
203                    node.activation = live_claim.activation_weight;
204                }
205                all_graph_nodes.push(node);
206            }
207        }
208    }
209
210    // 5. Merge vector search results with graph diffusion results
211    let merge_config = MergeConfig {
212        vector_weight: 0.6,
213        graph_weight: 0.4,
214        min_score: 0.1,
215        max_results: req.context.max_results as usize,
216    };
217
218    let claim_tuples: Vec<(uuid::Uuid, EpistemicClaim)> =
219        claims.iter().map(|c| (c.id, c.clone())).collect();
220
221    let (merged_results, related_by_graph) = merge_results(
222        &vector_matches,
223        &all_graph_nodes,
224        &claim_tuples,
225        &merge_config,
226    );
227
228    // 6. Find best belief from merged results
229    let best_belief = if !merged_results.is_empty() {
230        let best = merged_results.first();
231        best.map(|r| {
232            let claim = claims.iter().find(|c| c.id == r.claim_id);
233            BeliefSummary {
234                claim_id: Some(r.claim_id),
235                content: strip_yaml_frontmatter(&r.content),
236                confidence: r.confidence,
237                based_on: claims.len(),
238                consolidation_stage: claim
239                    .map(|c| c.consolidation_stage)
240                    .unwrap_or(ConsolidationStage::FastTrack),
241                claim_ids: merged_results.iter().map(|r| r.claim_id).collect(),
242            }
243        })
244    } else {
245        None
246    };
247
248    // 7. Detect conflicts
249    let mut conflicts = Vec::new();
250    let mut sampling_conflicts = Vec::new();
251    if req.include_conflicts {
252        let mut seen_conflicts = std::collections::HashSet::new();
253        for claim in &claims {
254            let claim_conflicts = match claim_store
255                .get_conflicts_for_claim(claim.id, tenant_id)
256                .await
257            {
258                Ok(c) => c,
259                Err(e) => {
260                    tracing::warn!(claim_id = %claim.id, error = %e, "Conflict query failed");
261                    vec![]
262                }
263            };
264            for c in &claim_conflicts {
265                if !seen_conflicts.contains(&c.id) {
266                    seen_conflicts.insert(c.id);
267                    conflicts.push(ConflictSummary::from(c));
268
269                    if req.delegate_to_sampling {
270                        let claim_a_content = claims
271                            .iter()
272                            .find(|x| x.id == c.claim_a_id)
273                            .map(|x| x.content.clone())
274                            .unwrap_or_else(|| format!("Claim {}", c.claim_a_id));
275                        let claim_b_content = claims
276                            .iter()
277                            .find(|x| x.id == c.claim_b_id)
278                            .map(|x| x.content.clone())
279                            .unwrap_or_else(|| format!("Claim {}", c.claim_b_id));
280
281                        sampling_conflicts.push(crate::server::ConflictInfo {
282                            claim_a: claim_a_content,
283                            claim_b: claim_b_content,
284                            conflict_type: Some(format!("{:?}", c.conflict_type)),
285                        });
286                    }
287                }
288            }
289        }
290
291        // Delegate to sampling for conflict analysis if requested
292        if req.delegate_to_sampling
293            && !sampling_conflicts.is_empty()
294            && let Some(ref client) = llm_client
295        {
296            let knowledge_items: Vec<crate::server::KnowledgeItem> = claims
297                .iter()
298                .map(|c| crate::server::KnowledgeItem {
299                    id: c.id.to_string(),
300                    content: c.content.clone(),
301                    confidence: c.confidence as f32,
302                    source: None,
303                })
304                .collect();
305
306            let context = crate::server::SamplingContext {
307                knowledge_items,
308                conflicts: sampling_conflicts,
309                query_context: Some(req.query.clone()),
310                extra: std::collections::HashMap::new(),
311            };
312
313            let sampling_req = crate::server::SamplingRequest {
314                sampling_type: crate::server::SamplingType::ConflictAnalysis,
315                context,
316                prompt: "Analyze the conflicts detected in the knowledge base and provide insights"
317                    .to_string(),
318                max_tokens: 1024,
319            };
320
321            let client_ref: &dyn LlmClient = client.as_ref();
322            if let Ok(sampling_result) = call_sampling_protocol(client_ref, sampling_req).await {
323                for conflict in &mut conflicts {
324                    conflict.sampling_analysis = Some(sampling_result.content.clone());
325                }
326            }
327        }
328    }
329
330    // 7. Generate prediction if requested
331    let mut prediction = if req.include_predictions && !merged_results.is_empty() {
332        Some(generate_prediction(&req.query, &claims, &related_by_graph, llm_client.clone()).await)
333    } else {
334        None
335    };
336
337    // Delegate to sampling for prediction generation if requested
338    if req.delegate_to_sampling
339        && prediction.is_some()
340        && !claims.is_empty()
341        && let Some(ref client) = llm_client
342    {
343        let knowledge_items: Vec<crate::server::KnowledgeItem> = claims
344            .iter()
345            .map(|c| crate::server::KnowledgeItem {
346                id: c.id.to_string(),
347                content: c.content.clone(),
348                confidence: c.confidence as f32,
349                source: None,
350            })
351            .collect();
352
353        let context = crate::server::SamplingContext {
354            knowledge_items,
355            conflicts: vec![],
356            query_context: Some(req.query.clone()),
357            extra: std::collections::HashMap::new(),
358        };
359
360        let sampling_req = crate::server::SamplingRequest {
361            sampling_type: crate::server::SamplingType::PredictionGeneration,
362            context,
363            prompt: "Generate a more detailed prediction based on the knowledge base".to_string(),
364            max_tokens: 1024,
365        };
366
367        let client_ref: &dyn LlmClient = client.as_ref();
368        if let Ok(sampling_result) = call_sampling_protocol(client_ref, sampling_req).await
369            && let Some(ref mut pred) = prediction
370        {
371            pred.sampling_analysis = Some(sampling_result.content);
372        }
373    }
374
375    // 8. Detect knowledge gaps if requested (Issue #151: auto-identification)
376    let knowledge_gaps = if req.include_gaps && !merged_results.is_empty() {
377        detect_and_record_knowledge_gaps(
378            &req.query,
379            &claims,
380            &related_by_graph,
381            &req.context.domain,
382            &conflicts,
383            tenant_id,
384            gap_store,
385        )
386        .await
387    } else {
388        vec![]
389    };
390
391    // 9. Determine freshness
392    let newest_source = claims.iter().map(|c| c.t_known).max();
393    let oldest_source = claims.iter().map(|c| c.t_known).min();
394    let staleness_warning = claims.is_empty()
395        || newest_source
396            .map(|d| (chrono::Utc::now() - d).num_days() > 90)
397            .unwrap_or(false);
398
399    // 10. Build response
400    let execution_time_ms = start_time.elapsed().as_millis() as u64;
401    let related_node_count = related_by_graph.len();
402    let conflict_count = conflicts.len();
403
404    let response = McpQueryResponse {
405        query_hash,
406        query_context: req.query.clone(),
407        best_belief,
408        related_by_graph,
409        conflicts,
410        prediction,
411        knowledge_gaps,
412        freshness: FreshnessInfo {
413            newest_source,
414            oldest_source,
415            staleness_warning,
416        },
417        cache_status: CacheStatus::Miss,
418        cognitive_path: Some(CognitivePath::System2),
419        metadata: QueryMetadata {
420            execution_time_ms,
421            cache_hit_rate: 0.0,
422            processed_claims: claims.len(),
423            related_node_count,
424            conflict_count,
425        },
426    };
427
428    // 11. Update cache
429    let cache_entry = QueryCacheEntry::new(query_hash, response.clone());
430    cache_store.set_cached(tenant_id, &cache_entry).await?;
431
432    // 12. Update activation
433    for claim in &claims {
434        claim_store
435            .update_activation(claim.id, tenant_id, 0.1)
436            .await
437            .ok();
438    }
439
440    Ok(response)
441}

443/// Call sampling protocol for advanced LLM-based analysis
444pub(crate) async fn call_sampling_protocol(
445    llm_client: &dyn LlmClient,
446    sampling_req: crate::server::SamplingRequest,
447) -> std::result::Result<crate::server::SamplingResponse, JsonRpcError> {
448    use crate::server::{SamplingResponse, SamplingType};
449
450    let prompt = match sampling_req.sampling_type {
451        SamplingType::ConflictAnalysis => {
452            let context = &sampling_req.context;
453            let mut prompt = format!(
454                "You are an expert at analyzing conflicts between knowledge claims.\n\n\
455                Please analyze the following conflicting knowledge items and provide:\n\
456                1. A clear identification of the conflict\n\
457                2. Possible resolutions or reconciliation\n\
458                3. Recommended actions\n\n\
459                Query Context: {}\n\n",
460                context.query_context.as_deref().unwrap_or("N/A")
461            );
462
463            for (i, item) in context.knowledge_items.iter().enumerate() {
464                prompt.push_str(&format!(
465                    "\n--- Knowledge Item {} ---\nContent: {}\nConfidence: {:.2}\nSource: {}\n",
466                    i + 1,
467                    item.content,
468                    item.confidence,
469                    item.source.as_deref().unwrap_or("Unknown")
470                ));
471            }
472
473            if !context.conflicts.is_empty() {
474                prompt.push_str("\n--- Known Conflicts ---\n");
475                for conflict in &context.conflicts {
476                    prompt.push_str(&format!(
477                        "Claim A: {}\nClaim B: {}\nType: {}\n",
478                        conflict.claim_a,
479                        conflict.claim_b,
480                        conflict.conflict_type.as_deref().unwrap_or("Unknown")
481                    ));
482                }
483            }
484
485            prompt.push_str("\nPlease provide your analysis:");
486            prompt
487        }
488        SamplingType::KnowledgeValidation => {
489            let context = &sampling_req.context;
490            let mut prompt = String::from(
491                "You are an expert at validating knowledge claims.\n\n\
492                Please evaluate the following knowledge item(s) for:\n\
493                1. Factual accuracy\n\
494                2. Source reliability\n\
495                3. Confidence assessment\n\
496                4. Potential issues or concerns\n\n",
497            );
498
499            for (i, item) in context.knowledge_items.iter().enumerate() {
500                prompt.push_str(&format!(
501                    "\n--- Knowledge Item {} ---\nContent: {}\nConfidence: {:.2}\nSource: {}\n",
502                    i + 1,
503                    item.content,
504                    item.confidence,
505                    item.source.as_deref().unwrap_or("Unknown")
506                ));
507            }
508
509            if let Some(query_ctx) = &context.query_context {
510                prompt.push_str(&format!("\nQuery Context: {}\n", query_ctx));
511            }
512
513            prompt.push_str("\nPlease provide your validation assessment:");
514            prompt
515        }
516        SamplingType::PredictionGeneration => {
517            let context = &sampling_req.context;
518            let mut prompt = String::from(
519                "You are an expert at generating informed predictions based on knowledge.\n\n\
520                Based on the following knowledge items and context, please generate:\n\
521                1. Likely future outcomes\n\
522                2. Key factors influencing the prediction\n\
523                3. Confidence level and uncertainty factors\n\
524                4. Recommended monitoring points\n\n",
525            );
526
527            for (i, item) in context.knowledge_items.iter().enumerate() {
528                prompt.push_str(&format!(
529                    "\n--- Knowledge Item {} ---\nContent: {}\nConfidence: {:.2}\n",
530                    i + 1,
531                    item.content,
532                    item.confidence
533                ));
534            }
535
536            if let Some(query_ctx) = &context.query_context {
537                prompt.push_str(&format!("\nQuery/Task: {}\n", query_ctx));
538            }
539
540            prompt.push_str("\nPlease provide your prediction:");
541            prompt
542        }
543    };
544
545    let llm_req = LlmRequest {
546        messages: vec![Message {
547            role: Role::User,
548            content: prompt,
549        }],
550        temperature: 0.7,
551        max_tokens: Some(sampling_req.max_tokens),
552        ..Default::default()
553    };
554
555    let llm_response = llm_client
556        .chat(llm_req)
557        .await
558        .map_err(|e| JsonRpcError::new(-32000, format!("LLM call failed: {}", e)))?;
559
560    let content = llm_response.content;
561    let confidence = if content.is_empty() { 0.0 } else { 0.8 };
562
563    Ok(SamplingResponse::new(
564        &format!("{:?}", sampling_req.sampling_type).to_lowercase(),
565        content,
566        confidence,
567    ))
568}

570/// Generate prediction based on claims and graph relations
571async fn generate_prediction(
572    query: &str,
573    claims: &[EpistemicClaim],
574    related_graph: &[GraphRelation],
575    llm_client: Option<Arc<dyn LlmClient>>,
576) -> PredictionResult {
577    let avg_confidence: f64 = if !claims.is_empty() {
578        claims.iter().map(|c| c.confidence).sum::<f64>() / claims.len() as f64
579    } else {
580        0.5
581    };
582
583    if let Some(client) = llm_client {
584        let llm_prediction = generate_llm_prediction(
585            query,
586            claims,
587            related_graph,
588            client.as_ref(),
589            avg_confidence,
590        )
591        .await;
592
593        if let Some(prediction) = llm_prediction {
594            return prediction;
595        }
596    }
597
598    let content = if claims.len() < 3 {
599        format!(
600            "⚠️ 知识库中关于'{}'的信息有限({}条)，建议收集更多数据后决策",
601            query,
602            claims.len()
603        )
604    } else if avg_confidence > 0.7 {
605        format!(
606            "✅ 基于{}条高置信度知识，关于'{}'的决策可信度较高",
607            claims.len(),
608            query
609        )
610    } else if !related_graph.is_empty() {
611        format!("📊 发现{}条相关联知识，可作为决策参考", related_graph.len())
612    } else {
613        format!("💡 关于'{}'的知识已整合完毕，建议按现有信息执行", query)
614    };
615
616    PredictionResult {
617        content,
618        confidence: avg_confidence * 0.8,
619        method: PredictionMethod::StatisticalTrend,
620        based_on_claims: claims.iter().map(|c| c.id).collect(),
621        sampling_analysis: None,
622    }
623}

625/// Generate prediction using LLM
626async fn generate_llm_prediction(
627    query: &str,
628    claims: &[EpistemicClaim],
629    related_graph: &[GraphRelation],
630    client: &dyn LlmClient,
631    avg_confidence: f64,
632) -> Option<PredictionResult> {
633    let claims_context: String = claims
634        .iter()
635        .take(10)
636        .enumerate()
637        .map(|(i, c)| {
638            format!(
639                "{}. [置信度 {:.0}%] {}\n",
640                i + 1,
641                c.confidence * 100.0,
642                c.content.chars().take(200).collect::<String>()
643            )
644        })
645        .collect();
646
647    let graph_context: String = related_graph
648        .iter()
649        .take(5)
650        .enumerate()
651        .map(|(i, r)| {
652            format!(
653                "{}. {}: {}\n",
654                i + 1,
655                r.relation_type,
656                r.content.chars().take(100).collect::<String>()
657            )
658        })
659        .collect();
660
661    let system_prompt = r#"你是一个知识管理系统中的决策预测助手。
662你的任务是基于知识库中的相关信息，对用户的查询给出：
6631. 推荐建议
6642. 风险评估
6653. 执行建议
666
667请用中文回复，保持简洁（不超过100字）。
668如果信息不足，请明确指出不确定性和风险。"#;
669
670    let user_prompt = format!(
671        "用户查询: {}\n\n知识库中的相关信息:\n{}\n\n知识图谱关联:\n{}\n\n请给出预测建议（推荐/风险/建议）和置信度。",
672        query, claims_context, graph_context
673    );
674
675    let request = LlmRequest {
676        model: "gpt-4o-mini".to_string(),
677        messages: vec![
678            Message {
679                role: Role::System,
680                content: system_prompt.to_string(),
681            },
682            Message {
683                role: Role::User,
684                content: user_prompt,
685            },
686        ],
687        temperature: 0.4,
688        max_tokens: Some(200),
689        top_p: None,
690        stop_sequences: vec![],
691    };
692
693    match client.chat(request).await {
694        Ok(response) => {
695            let confidence = (avg_confidence * 0.7 + 0.3).min(1.0);
696
697            Some(PredictionResult {
698                content: response.content,
699                confidence,
700                method: PredictionMethod::LlmBeliefContext,
701                based_on_claims: claims.iter().map(|c| c.id).collect(),
702                sampling_analysis: None,
703            })
704        }
705        Err(e) => {
706            tracing::warn!("LLM prediction failed: {}, falling back to rule-based", e);
707            None
708        }
709    }
710}

712/// Detect knowledge gaps based on query and available claims
713fn detect_knowledge_gaps(
714    query: &str,
715    claims: &[EpistemicClaim],
716    related_graph: &[GraphRelation],
717    domain: &Option<String>,
718    conflicts: &[ConflictSummary],
719) -> Vec<String> {
720    let mut gaps = Vec::new();
721
722    if claims.is_empty() {
723        gaps.push(format!("完全缺失关于'{}'的知识储备", query));
724        return gaps;
725    }
726
727    if claims.len() < 3 {
728        gaps.push(format!(
729            "知识稀疏：当前仅有{}条相关记录，不足以进行多方验证",
730            claims.len()
731        ));
732    }
733
734    let avg_confidence: f64 =
735        claims.iter().map(|c| c.confidence).sum::<f64>() / claims.len() as f64;
736    if avg_confidence < 0.5 {
737        gaps.push(format!(
738            "置信度缺口：平均置信度仅为{:.2}，建议补充权威数据源",
739            avg_confidence
740        ));
741    }
742
743    if related_graph.is_empty() && claims.len() > 2 {
744        gaps.push("关联缺口：发现孤立知识点，知识库中缺乏逻辑连接".to_string());
745    }
746
747    let now = chrono::Utc::now();
748    let newest = match claims.iter().map(|c| c.t_known).max() {
749        Some(t) => t,
750        None => return gaps,
751    };
752    let age_days = (now - newest).num_days();
753    if age_days > 30 {
754        gaps.push(format!(
755            "时效性缺口：最新知识已是{}天前，可能已过时",
756            age_days
757        ));
758    }
759
760    if claims.len() >= 3 && conflicts.len() as f64 / claims.len() as f64 > 1.0 {
761        gaps.push("认知冲突：该领域存在大量矛盾信息，知识一致性极低".to_string());
762    }
763
764    if let Some(d) = domain
765        && claims.len() < 5
766    {
767        gaps.push(format!("领域缺口：'{}'领域深度不足，建议定向采集", d));
768    }
769
770    gaps
771}

773/// Detect and automatically record knowledge gaps to the GapStore
774async fn detect_and_record_knowledge_gaps(
775    query: &str,
776    claims: &[EpistemicClaim],
777    related_graph: &[GraphRelation],
778    domain: &Option<String>,
779    conflicts: &[ConflictSummary],
780    tenant_id: &str,
781    gap_store: &dyn GapStore,
782) -> Vec<String> {
783    let gaps = detect_knowledge_gaps(query, claims, related_graph, domain, conflicts);
784
785    for gap_description in &gaps {
786        let priority = if gap_description.contains("完全缺失")
787            || gap_description.contains("置信度缺口")
788        {
789            "high"
790        } else if gap_description.contains("领域缺口") || gap_description.contains("知识稀疏")
791        {
792            "medium"
793        } else {
794            "low"
795        };
796
797        let gap_record = KnowledgeGapRecord {
798            gap_id: uuid::Uuid::new_v4(),
799            tenant_id: tenant_id.to_string(),
800            domain: domain.clone().unwrap_or_else(|| "unclassified".to_string()),
801            description: gap_description.clone(),
802            priority: priority.to_string(),
803            status: "open".to_string(),
804            reported_at: chrono::Utc::now(),
805            filled_at: None,
806        };
807
808        if let Err(e) = gap_store.record_gap(&gap_record).await {
809            tracing::warn!("Failed to auto-record knowledge gap: {}", e);
810        }
811    }
812
813    gaps
814}

816/// Strip YAML frontmatter (---\n...\n---) from content
817fn strip_yaml_frontmatter(content: &str) -> String {
818    let trimmed = content.trim();
819    if trimmed.starts_with("---") {
820        if let Some(end_pos) = trimmed[3..].find("\n---") {
821            let after = &trimmed[3 + end_pos + 4..];
822            return after.trim_start_matches('\n').to_string();
823        }
824    }
825    content.to_string()
826}
