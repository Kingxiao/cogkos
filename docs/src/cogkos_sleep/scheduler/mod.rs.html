20/// Task types for budget allocation
21#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
22pub enum TaskType {
23    /// Event-driven conflict detection (on write)
24    ConflictDetection,
25    /// Event-driven consolidation (triggered by novel knowledge from ingest)
26    ConsolidationEventDriven,
27    /// Periodic consolidation (every 6h)
28    Consolidation,
29    /// Periodic decay (daily)
30    Decay,
31    /// Periodic health check (hourly)
32    HealthCheck,
33    /// Periodic conflict detection
34    ConflictDetectionPeriodic,
35    /// Confidence boost for similar knowledge (triggered by Sleep-time scheduler)
36    ConfidenceBoost,
37    /// Working memory GC (expire old working/episodic claims)
38    MemoryGc,
39    /// Memory layer promotion (working → episodic → semantic)
40    MemoryPromotion,
41}
42
43impl TaskType {
44    pub fn name(&self) -> &'static str {
45        match self {
46            TaskType::ConflictDetection => "conflict_detection",
47            TaskType::ConsolidationEventDriven => "consolidation_event_driven",
48            TaskType::Consolidation => "consolidation",
49            TaskType::Decay => "decay",
50            TaskType::HealthCheck => "health_check",
51            TaskType::ConflictDetectionPeriodic => "conflict_detection_periodic",
52            TaskType::ConfidenceBoost => "confidence_boost",
53            TaskType::MemoryGc => "memory_gc",
54            TaskType::MemoryPromotion => "memory_promotion",
55        }
56    }
57}

59/// Scheduler configuration
60#[derive(Debug, Clone)]
61pub struct SchedulerConfig {
62    /// Consolidation interval in seconds
63    pub consolidation_interval_secs: u64,
64    /// Event-driven consolidation interval in seconds (for PendingAggregation claims)
65    pub consolidation_event_driven_interval_secs: u64,
66    /// Decay interval in seconds
67    pub decay_interval_secs: u64,
68    /// Health check interval in seconds
69    pub health_check_interval_secs: u64,
70    /// Conflict detection interval in seconds
71    pub conflict_interval_secs: u64,
72    /// Confidence boost interval in seconds
73    pub confidence_boost_interval_secs: u64,
74    /// Conflict detection batch size
75    pub conflict_batch_size: usize,
76    /// Confidence boost batch size
77    pub confidence_boost_batch_size: usize,
78    /// Confidence boost factor (how much to boost each time)
79    pub confidence_boost_factor: f64,
80    /// Memory GC interval in seconds
81    pub memory_gc_interval_secs: u64,
82    /// Memory promotion interval in seconds
83    pub memory_promotion_interval_secs: u64,
84    /// Min rehearsal count to promote working → episodic
85    pub working_to_episodic_rehearsal: u64,
86    /// Min rehearsal count to promote episodic → semantic
87    pub episodic_to_semantic_rehearsal: u64,
88    /// Whether to enable periodic tasks
89    pub enable_periodic: bool,
90    /// Maximum claims to process across all tasks per hour (budget)
91    pub max_claims_per_hour: u64,
92    /// Maximum time in milliseconds for a single background task cycle
93    pub max_task_duration_ms: u64,
94    /// Budget allocation percentages per task type (must sum to 100)
95    /// Key is task type name, value is percentage (0-100)
96    pub task_budget_percentages: HashMap<String, u8>,
97}
98
99impl Default for SchedulerConfig {
100    fn default() -> Self {
101        let mut task_budget_percentages = HashMap::new();
102        // Default budget allocation:
103        // - Consolidation Event-Driven: 20% (triggered by novel knowledge)
104        // - Consolidation: 15% (periodic)
105        // - Decay: 15%
106        // - Conflict Detection Periodic: 15%
107        // - Conflict Detection (event-driven): 5%
108        // - Confidence Boost: 10% (for similar knowledge aggregation)
109        // - Memory GC: 5% (expire working/episodic claims)
110        // - Memory Promotion: 5% (promote across layers)
111        // - Health Check: 5% (minimal work)
112        task_budget_percentages.insert(TaskType::ConsolidationEventDriven.name().to_string(), 20);
113        task_budget_percentages.insert(TaskType::Consolidation.name().to_string(), 15);
114        task_budget_percentages.insert(TaskType::Decay.name().to_string(), 15);
115        task_budget_percentages.insert(TaskType::ConflictDetectionPeriodic.name().to_string(), 10);
116        task_budget_percentages.insert(TaskType::ConflictDetection.name().to_string(), 5);
117        task_budget_percentages.insert(TaskType::ConfidenceBoost.name().to_string(), 10);
118        task_budget_percentages.insert(TaskType::MemoryGc.name().to_string(), 5);
119        task_budget_percentages.insert(TaskType::MemoryPromotion.name().to_string(), 5);
120        task_budget_percentages.insert(TaskType::HealthCheck.name().to_string(), 5);
121
122        Self {
123            consolidation_interval_secs: 6 * 3600,         // 6 hours
124            consolidation_event_driven_interval_secs: 300, // 5 minutes
125            decay_interval_secs: 24 * 3600,                // 24 hours
126            health_check_interval_secs: 3600,              // 1 hour
127            conflict_interval_secs: 2 * 3600,              // 2 hours
128            confidence_boost_interval_secs: 1800,          // 30 minutes
129            conflict_batch_size: 100,
130            confidence_boost_batch_size: 50,
131            confidence_boost_factor: 0.05,
132            memory_gc_interval_secs: 1800,        // 30 minutes
133            memory_promotion_interval_secs: 3600, // 1 hour
134            working_to_episodic_rehearsal: 3,     // 3 recalls to promote
135            episodic_to_semantic_rehearsal: 5,    // 5 recalls to promote
136            enable_periodic: true,
137            max_claims_per_hour: 10000,
138            max_task_duration_ms: 300_000, // 5 minutes
139            task_budget_percentages,
140        }
141    }
142}

144/// Scheduler state
145#[derive(Debug, Clone)]
146pub struct SchedulerState {
147    pub is_running: bool,
148    pub last_consolidation: Option<chrono::DateTime<chrono::Utc>>,
149    pub last_decay: Option<chrono::DateTime<chrono::Utc>>,
150    pub last_conflict_detection: Option<chrono::DateTime<chrono::Utc>>,
151    pub last_confidence_boost: Option<chrono::DateTime<chrono::Utc>>,
152    pub tasks_processed: u64,
153    pub total_claims_processed: u64,
154    pub errors: u64,
155    pub current_budget_usage: u64,
156    /// Per-task-type budget usage
157    pub task_budget_usage: HashMap<String, u64>,
158    pub last_budget_reset: chrono::DateTime<chrono::Utc>,
159}
160

163        Self {

165            last_consolidation: None,
166            last_decay: None,
167            last_conflict_detection: None,
168            last_confidence_boost: None,
169            tasks_processed: 0,
170            total_claims_processed: 0,
171            errors: 0,
172            current_budget_usage: 0,
173            task_budget_usage: HashMap::new(),
174            last_budget_reset: chrono::Utc::now(),
175        }
176    }
177}
