3//! Detects conflicts between newly written claims and existing knowledge.

12/// Configuration for conflict detection
13#[derive(Debug, Clone)]
14pub struct ConflictDetectionConfig {
15    /// Maximum number of existing claims to compare against
16    pub max_comparison_batch: usize,
17    /// Similarity threshold for conflict detection
18    pub similarity_threshold: f64,
19    /// Whether to use semantic similarity (requires LLM in Phase 3)
20    pub use_semantic: bool,
21}
22
23impl Default for ConflictDetectionConfig {
24    fn default() -> Self {
25        Self {
26            max_comparison_batch: 100,
27            similarity_threshold: 0.6,
28            use_semantic: false,
29        }
30    }
31}

33/// Run conflict detection for a newly written claim
34///
35/// This is event-driven - called after each write to the knowledge base
36#[tracing::instrument(skip(stores, new_claim, config), fields(claim_id = %new_claim.id, tenant = %new_claim.tenant_id))]
37pub async fn detect_conflicts(
38    stores: &Arc<Stores>,
39    new_claim: &EpistemicClaim,
40    config: &ConflictDetectionConfig,
41) -> Result<Vec<ConflictRecord>> {
42    info!(
43        claim_id = %new_claim.id,
44        tenant_id = %new_claim.tenant_id,
45        "Running conflict detection for new claim"
46    );
47
48    // Get existing claims to compare against
49    // For efficiency, we limit to recent/modified claims
50    let existing_claims = stores
51        .claims
52        .search_claims(
53            &new_claim.tenant_id,
54            &new_claim.content,
55            config.max_comparison_batch,
56        )
57        .await?;
58
59    debug!(
60        claim_id = %new_claim.id,
61        compared_against = existing_claims.len(),
62        "Retrieved existing claims for comparison"
63    );
64
65    // Detect conflicts
66    let conflicts = detect_conflicts_batch(new_claim, &existing_claims);
67
68    if conflicts.is_empty() {
69        debug!(claim_id = %new_claim.id, "No conflicts detected");
70        return Ok(vec![]);
71    }
72
73    info!(
74        claim_id = %new_claim.id,
75        conflict_count = conflicts.len(),
76        "Detected conflicts"
77    );
78
79    // Store conflicts in the database
80    for mut conflict in conflicts.iter().cloned() {
81        // Add description based on conflict type
82        conflict.description = Some(describe_conflict(&conflict.conflict_type));
83
84        if let Err(e) = stores.claims.insert_conflict(&conflict).await {
85            error!(error = %e, conflict_id = %conflict.id, "Failed to store conflict record");
86        }
87    }
88
89    Ok(conflicts)
90}

92/// Detect conflicts for all claims in a tenant (periodic task)
93pub async fn detect_conflicts_periodic(
94    stores: &Arc<Stores>,
95    tenant_id: &str,
96    limit: usize,
97) -> Result<Vec<ConflictRecord>> {
98    info!(tenant_id = %tenant_id, "Running periodic conflict detection");
99
100    // Get FastTrack claims (newest assertions)
101    let claims = stores
102        .claims
103        .list_claims_by_stage(
104            tenant_id,
105            cogkos_core::models::ConsolidationStage::FastTrack,
106            limit,
107        )
108        .await?;
109
110    let mut all_conflicts = Vec::new();
111
112    for claim in claims {
113        // Get claims to compare (exclude the current one)
114        let existing = stores
115            .claims
116            .search_claims(tenant_id, &claim.content, 20)
117            .await?;
118
119        let conflicts = detect_conflicts_batch(&claim, &existing);
120
121        for mut conflict in conflicts {
122            conflict.description = Some(describe_conflict(&conflict.conflict_type));
123
124            if let Err(e) = stores.claims.insert_conflict(&conflict).await {
125                warn!(error = %e, "Failed to store conflict");
126            } else {
127                all_conflicts.push(conflict);
128            }
129        }
130    }
131
132    info!(
133        tenant_id = %tenant_id,
134        total_conflicts = all_conflicts.len(),
135        "Periodic conflict detection complete"
136    );
137
138    Ok(all_conflicts)
139}

141/// Calculate conflict severity based on claim properties
142pub fn calculate_conflict_severity(claim_a: &EpistemicClaim, claim_b: &EpistemicClaim) -> f64 {
143    // Base severity from confidence difference
144    let confidence_diff = (claim_a.confidence - claim_b.confidence).abs();
145
146    // Higher confidence claims in conflict = more severe
147    let avg_confidence = (claim_a.confidence + claim_b.confidence) / 2.0;
148
149    // Source independence increases severity (different sources = more important)
150    let source_independence = if claim_a.provenance.source_id != claim_b.provenance.source_id {
151        1.0
152    } else {
153        0.5
154    };
155
156    // Combine factors
157    let severity = (1.0 - confidence_diff) * avg_confidence * source_independence;
158    severity.clamp(0.0, 1.0)
159}

161/// Resolve a detected conflict
162#[tracing::instrument(skip(stores))]
163pub async fn resolve_conflict(
164    stores: &Arc<Stores>,
165    conflict_id: uuid::Uuid,
166    tenant_id: &str,
167    resolution: ResolutionStatus,
168    note: Option<String>,
169) -> Result<()> {
170    info!(conflict_id = %conflict_id, resolution = ?resolution, "Resolving conflict");
171
172    stores
173        .claims
174        .resolve_conflict(conflict_id, tenant_id, resolution, note)
175        .await?;
176
177    info!(conflict_id = %conflict_id, "Conflict resolved");
178    Ok(())
179}

181/// Generate human-readable description of conflict type
182fn describe_conflict(conflict_type: &ConflictType) -> String {
183    match conflict_type {
184        ConflictType::DirectContradiction => "Direct contradiction between two claims".to_string(),
185        ConflictType::ContextDependent => "Conflict depends on context or scope".to_string(),
186        ConflictType::TemporalShift => {
187            "Temporal inconsistency - same entity, different time periods".to_string()
188        }
189        ConflictType::TemporalInconsistency => {
190            "Temporal inconsistency in validity periods".to_string()
191        }
192        ConflictType::SourceDisagreement => {
193            "Different sources provide conflicting information".to_string()
194        }
195        ConflictType::ConfidenceMismatch => {
196            "Significant confidence discrepancy between similar claims".to_string()
197        }
198        ConflictType::ContextualDifference => {
199            "Claims valid in different contexts but conflict".to_string()
200        }
201    }
202}

204/// Detect conflicts using LLM semantic analysis
205///
206/// Sends both claims to an LLM to determine if they semantically conflict,
207/// catching subtleties that rule-based detection misses.
208pub async fn detect_llm_semantic_conflict(
209    claim_a: &EpistemicClaim,
210    claim_b: &EpistemicClaim,
211    llm_client: &dyn cogkos_llm::LlmClient,
212) -> Option<ConflictType> {
213    use cogkos_llm::types::{LlmRequest, Message, Role};
214
215    let system_prompt = "You are a conflict detection system. Given two knowledge claims, determine if they contradict each other.\n\
216        Respond with EXACTLY one of these labels on the first line:\n\
217        - NONE — no conflict\n\
218        - DIRECT_CONTRADICTION — the claims directly contradict\n\
219        - TEMPORAL_SHIFT — the claims differ due to time\n\
220        - CONTEXT_DEPENDENT — the claims address different scopes but appear conflicting\n\
221        - CONFIDENCE_MISMATCH — same topic, very different confidence levels\n\
222        - SOURCE_DISAGREEMENT — same topic from incompatible sources\n\
223        - CONTEXTUAL_DIFFERENCE — meaning has shifted subtly\n\n\
224        On the second line, provide a brief reason (one sentence).";
225
226    let user_prompt = format!(
227        "Claim A (confidence {:.2}): {}\n\nClaim B (confidence {:.2}): {}",
228        claim_a.confidence, claim_a.content, claim_b.confidence, claim_b.content
229    );
230
231    let request = LlmRequest {
232        messages: vec![
233            Message {
234                role: Role::System,
235                content: system_prompt.to_string(),
236            },
237            Message {
238                role: Role::User,
239                content: user_prompt,
240            },
241        ],
242        temperature: 0.0,
243        max_tokens: Some(100),
244        ..Default::default()
245    };
246
247    match llm_client.chat(request).await {
248        Ok(response) => {
249            let first_line = response.content.lines().next().unwrap_or("").trim();
250            match first_line {
251                "DIRECT_CONTRADICTION" => Some(ConflictType::DirectContradiction),
252                "TEMPORAL_SHIFT" => Some(ConflictType::TemporalShift),
253                "CONTEXT_DEPENDENT" => Some(ConflictType::ContextDependent),
254                "CONFIDENCE_MISMATCH" => Some(ConflictType::ConfidenceMismatch),
255                "SOURCE_DISAGREEMENT" => Some(ConflictType::SourceDisagreement),
256                "CONTEXTUAL_DIFFERENCE" => Some(ConflictType::ContextualDifference),
257                "TEMPORAL_INCONSISTENCY" => Some(ConflictType::TemporalInconsistency),
258                _ => None,
259            }
260        }
261        Err(e) => {
262            warn!("LLM semantic conflict detection failed: {}", e);
263            None
264        }
265    }
266}

268/// Enhanced conflict detection that combines rule-based and LLM-based analysis
269pub async fn detect_conflicts_enhanced(
270    stores: &Arc<Stores>,
271    new_claim: &EpistemicClaim,
272    config: &ConflictDetectionConfig,
273    llm_client: Option<&dyn cogkos_llm::LlmClient>,
274) -> Result<Vec<ConflictRecord>> {
275    // First run rule-based detection
276    let mut conflicts = detect_conflicts(stores, new_claim, config).await?;
277
278    // If LLM semantic detection is enabled and we have a client
279    if config.use_semantic
280        && let Some(client) = llm_client
281    {
282        let existing_claims = stores
283            .claims
284            .search_claims(
285                &new_claim.tenant_id,
286                &new_claim.content,
287                config.max_comparison_batch,
288            )
289            .await?;
290
291        // Only run LLM on pairs not already flagged by rule-based detection
292        let flagged_ids: std::collections::HashSet<_> = conflicts
293            .iter()
294            .flat_map(|c| vec![c.claim_a_id, c.claim_b_id])
295            .collect();
296
297        for existing in &existing_claims {
298            if existing.id == new_claim.id || flagged_ids.contains(&existing.id) {
299                continue;
300            }
301
302            if let Some(conflict_type) =
303                detect_llm_semantic_conflict(new_claim, existing, client).await
304            {
305                let mut record = ConflictRecord::new(
306                    new_claim.tenant_id.clone(),
307                    new_claim.id,
308                    existing.id,
309                    conflict_type,
310                );
311                record.description = Some(describe_conflict(&conflict_type));
312                record.severity = calculate_conflict_severity(new_claim, existing);
313
314                if let Err(e) = stores.claims.insert_conflict(&record).await {
315                    error!(error = %e, "Failed to store LLM-detected conflict");
316                } else {
317                    conflicts.push(record);
318                }
319            }
320        }
321    }
322
323    Ok(conflicts)
324}

326/// Get open conflicts for a tenant
327pub async fn get_open_conflicts(
328    stores: &Arc<Stores>,
329    tenant_id: &str,
330) -> Result<Vec<ConflictRecord>> {
331    // Query claims and filter for those with conflicts
332    // This is a simplified implementation
333    let claims = stores
334        .claims
335        .list_claims_by_stage(
336            tenant_id,
337            cogkos_core::models::ConsolidationStage::FastTrack,
338            100,
339        )
340        .await?;
341
342    let mut conflicts = Vec::new();
343    for claim in claims {
344        let claim_conflicts = stores
345            .claims
346            .get_conflicts_for_claim(claim.id, tenant_id)
347            .await?;
348        for conflict in claim_conflicts {
349            if conflict.resolution_status == ResolutionStatus::Open {
350                conflicts.push(conflict);
351            }
352        }
353    }
354
355    Ok(conflicts)
356}
