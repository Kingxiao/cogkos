13/// Federation protocol errors
14#[derive(Error, Debug)]
15pub enum FederationProtocolError {
16    #[error("Export failed: {0}")]
17    ExportFailed(String),
18    #[error("Import failed: {0}")]
19    ImportFailed(String),
20    #[error("Authentication failed: {0}")]
21    AuthenticationFailed(String),
22    #[error("Protocol error: {0}")]
23    ProtocolError(String),
24    #[error("Serialization error: {0}")]
25    SerializationError(#[from] serde_json::Error),
26    #[error("IO error: {0}")]
27    IoError(#[from] std::io::Error),
28    #[error("HTTP error: {0}")]
29    HttpError(String),
30}
31
32pub type Result<T> = std::result::Result<T, FederationProtocolError>;

55/// Time bucketing for anonymity
56#[derive(Debug, Clone, Copy, Serialize, Deserialize)]
57pub enum TimeBucket {
58    Hour,
59    Day,
60    Week,
61    Month,
62    Quarter,
63    Year,
64}
65
66impl TimeBucket {
67    /// Convert timestamp to bucket
68    pub fn from_timestamp(timestamp: DateTime<Utc>, bucket: Self) -> String {
69        match bucket {
70            TimeBucket::Hour => timestamp.format("%Y-%m-%d-%H").to_string(),
71            TimeBucket::Day => timestamp.format("%Y-%m-%d").to_string(),
72            TimeBucket::Week => {
73                let week = timestamp.iso_week();
74                format!("{}-W{:02}", timestamp.year(), week.week())
75            }
76            TimeBucket::Month => timestamp.format("%Y-%m").to_string(),
77            TimeBucket::Quarter => {
78                let quarter = (timestamp.month() - 1) / 3 + 1;
79                format!("{}-Q{}", timestamp.year(), quarter)
80            }
81            TimeBucket::Year => timestamp.format("%Y").to_string(),
82        }
83    }
84}

114/// Anonymization configuration
115#[derive(Debug, Clone)]
116pub struct AnonymizationConfig {
117    /// Time bucket granularity
118    pub time_bucket: TimeBucket,
119    /// Minimum confidence threshold for export
120    pub min_confidence: f64,
121    /// Required consolidation stage
122    pub min_stage: ConsolidationStage,
123    /// Anonymize entity names
124    pub anonymize_entities: bool,
125    /// Remove geographic references
126    pub remove_locations: bool,
127    /// Hash algorithm for IDs

129}

151/// Insight anonymizer
152pub struct InsightAnonymizer {
153    config: AnonymizationConfig,
154    entity_patterns: Vec<regex::Regex>,
155    location_patterns: Vec<regex::Regex>,
156}
157
158impl InsightAnonymizer {
159    pub fn new(config: AnonymizationConfig) -> Self {
160        let entity_patterns = vec![
161            regex::Regex::new(r"\b[A-Z][a-z]+ (Inc|Corp|Ltd|LLC|Company)\b").expect("valid regex"),
162            regex::Regex::new(r"\b[A-Z][a-z]+ [A-Z][a-z]+\b").expect("valid regex"),
163        ];
164
165        let location_patterns =
166            vec![regex::Regex::new(r"\b(?:in|at|near) [A-Z][a-zA-Z\s]+\b").expect("valid regex")];
167
168        Self {
169            config,
170            entity_patterns,
171            location_patterns,
172        }
173    }
174
175    /// Anonymize a single claim
176    pub fn anonymize(&self, claim: &EpistemicClaim, instance_id: &str) -> Result<AnonymousInsight> {
177        // Check thresholds
178        if claim.confidence < self.config.min_confidence {
179            return Err(FederationProtocolError::ExportFailed(format!(
180                "Confidence {} below threshold {}",
181                claim.confidence, self.config.min_confidence
182            )));
183        }
184
185        if (claim.consolidation_stage as i32) < (self.config.min_stage as i32) {
186            return Err(FederationProtocolError::ExportFailed(
187                "Consolidation stage below threshold".to_string(),
188            ));
189        }
190
191        // Anonymize content
192        let anonymized_content = self.anonymize_content(&claim.content);
193
194        // Generate content hash
195        let content_hash = self.hash_content(&anonymized_content);
196
197        // Generate source instance hash
198        let source_instance_hash = self.hash_instance(instance_id);
199
200        // Calculate statistics
201        let statistics = InsightStatistics {
202            support_count: 1, // Would be calculated from claim relationships
203            conflict_count: 0,
204            source_diversity: 0.5, // Calculated from source types
205            temporal_range_days: 30,
206            normalized_activation: claim.activation_weight,
207        };
208
209        let validation = ValidationMetadata {
210            successful_predictions: 0,
211            failed_predictions: 0,
212            last_validated_bucket: TimeBucket::from_timestamp(Utc::now(), self.config.time_bucket),
213            corroboration_count: 0,
214        };
215
216        Ok(AnonymousInsight {
217            id: Uuid::new_v4(),
218            content_hash,
219            anonymized_content,
220            node_type: claim.node_type,
221            confidence: claim.confidence,
222            consolidation_stage: claim.consolidation_stage,
223            domain_tags: claim
224                .metadata
225                .get("domain")
226                .and_then(|v| v.as_array())
227                .map(|arr| {
228                    arr.iter()
229                        .filter_map(|v| v.as_str().map(String::from))
230                        .collect()
231                })
232                .unwrap_or_default(),
233            time_bucket: self.config.time_bucket,
234            source_instance_hash,
235            statistics,
236            validation,
237        })
238    }
239
240    /// Anonymize content by removing/replacing identifying information
241    pub(crate) fn anonymize_content(&self, content: &str) -> String {
242        let mut result = content.to_string();
243
244        if self.config.anonymize_entities {
245            for pattern in &self.entity_patterns {
246                result = pattern.replace_all(&result, "[ENTITY]").to_string();
247            }
248        }
249
250        if self.config.remove_locations {
251            for pattern in &self.location_patterns {
252                result = pattern.replace_all(&result, "[LOCATION]").to_string();
253            }
254        }
255
256        result
257    }
258
259    /// Hash content for deduplication
260    fn hash_content(&self, content: &str) -> String {
261        use std::collections::hash_map::DefaultHasher;
262        use std::hash::{Hash, Hasher};
263
264        let mut hasher = DefaultHasher::new();
265        content.hash(&mut hasher);
266        format!("{:x}", hasher.finish())
267    }
268
269    /// Hash instance ID
270    fn hash_instance(&self, instance_id: &str) -> String {
271        self.hash_content(instance_id)
272    }
273
274    /// Batch anonymize claims
275    pub fn anonymize_batch(
276        &self,
277        claims: &[EpistemicClaim],
278        instance_id: &str,
279    ) -> Vec<AnonymousInsight> {
280        claims
281            .iter()
282            .filter_map(|claim| self.anonymize(claim, instance_id).ok())
283            .collect()
284    }
